204class TreeItemComponent : public Component
205{
206public:
207 TreeItemComponent (JucerTreeViewBase& i) : item (&i)
208 {
209 setAccessible (false);
210 setInterceptsMouseClicks (false, true);
211 item->textX = iconWidth;
212 }
213
214 void paint (Graphics& g) override
215 {
216 if (item == nullptr)
217 return;
218
219 auto bounds = getLocalBounds().toFloat();
220 auto iconBounds = bounds.removeFromLeft ((float) iconWidth).reduced (7, 5);
221
222 bounds.removeFromRight ((float) buttons.size() * bounds.getHeight());
223
224 item->paintIcon (g, iconBounds);
225 item->paintContent (g, bounds.toNearestInt());
226 }
227
228 void resized() override
229 {
230 auto r = getLocalBounds();
231
232 for (int i = buttons.size(); --i >= 0;)
233 buttons.getUnchecked (i)->setBounds (r.removeFromRight (r.getHeight()));
234 }
235
236 void addRightHandButton (Component* button)
237 {
238 buttons.add (button);
239 addAndMakeVisible (button);
240 }
241
242 WeakReference<JucerTreeViewBase> item;
243 OwnedArray<Component> buttons;
244
245 const int iconWidth = 25;
246
247 JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR (TreeItemComponent)
248};
