107inline juce::int32 b2DistanceProxy::GetSupport(const b2Vec2& d) const
108{
109 juce::int32 bestIndex = 0;
110 float32 bestValue = b2Dot(m_vertices[0], d);
111 for (juce::int32 i = 1; i < m_count; ++i)
112 {
113 float32 value = b2Dot(m_vertices[i], d);
114 if (value > bestValue)
115 {
116 bestIndex = i;
117 bestValue = value;
118 }
119 }
120
121 return bestIndex;
122}

124inline const b2Vec2& b2DistanceProxy::GetSupportVertex(const b2Vec2& d) const
125{
126 juce::int32 bestIndex = 0;

128 for (juce::int32 i = 1; i < m_count; ++i)

131 if (value > bestValue)
132 {
133 bestIndex = i;
134 bestValue = value;
135 }
136 }
