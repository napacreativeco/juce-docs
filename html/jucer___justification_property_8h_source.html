61 void setIndex (int newIndex)
62 {
63 const int types[] = { Justification::centred,
64 Justification::centredLeft,
65 Justification::centredRight,
66 Justification::centredTop,
67 Justification::centredBottom,
68 Justification::topLeft,
69 Justification::topRight,
70 Justification::bottomLeft,
71 Justification::bottomRight };
72
73 if (((unsigned int) newIndex) < (unsigned int) numElementsInArray (types)
74 && types [newIndex] != getJustification().getFlags())
75 {
76 setJustification (Justification (types [newIndex]));
77 }
78 }

80 int getIndex() const
81 {
82 const int types[] = { Justification::centred,
83 Justification::centredLeft,
84 Justification::centredRight,
85 Justification::centredTop,
86 Justification::centredBottom,
87 Justification::topLeft,
88 Justification::topRight,
89 Justification::bottomLeft,
90 Justification::bottomRight };
91
92 const int rawFlags = getJustification().getFlags();
93
94 for (int i = numElementsInArray (types); --i >= 0;)
95 if (types[i] == rawFlags)
96 return i;
97
98 return -1;
99 }
