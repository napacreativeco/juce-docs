96 struct Type : public OpenDocumentManager::DocumentType
97 {
98 bool canOpenFile (const File& file) override
99 {
100 if (file.hasFileExtension (sourceOrHeaderFileExtensions)
101 || file.hasFileExtension ("txt;inc;tcc;xml;plist;rtf;html;htm;php;py;rb;cs"))
102 return true;
103
104 MemoryBlock mb;
105 if (file.loadFileAsData (mb)
106 && seemsToBeText (static_cast<const char*> (mb.getData()), (int) mb.getSize())
107 && ! file.hasFileExtension ("svg"))
108 return true;
109
110 return false;
111 }
112
113 static bool seemsToBeText (const char* const chars, const int num) noexcept
114 {
115 for (int i = 0; i < num; ++i)
116 {
117 const char c = chars[i];
118 if ((c < 32 && c != '\t' && c != '\r' && c != '\n') || chars[i] > 126)
119 return false;
120 }
121
122 return true;
123 }
124
125 Document* openFile (Project* p, const File& file) override { return new SourceCodeDocument (p, file); }
126 };
