100// We want to send type-erased commands to the audio thread, but we also
101// want those commands to contain move-only resources, so that we can
102// construct resources on the gui thread, and then transfer ownership
103// cheaply to the audio thread. We can't do this with std::function
104// because it enforces that functions are copy-constructible.
105// Therefore, we use a very simple templated type-eraser here.
106template <typename Proc>
107struct Command
108{
109 virtual ~Command() noexcept = default;
110 virtual void run (Proc& proc) = 0;
111};
112
113template <typename Proc, typename Func>
114class TemplateCommand : public Command<Proc>,
115 private Func
116{
117public:
118 template <typename FuncPrime>
119 explicit TemplateCommand (FuncPrime&& funcPrime)
120 : Func (std::forward<FuncPrime> (funcPrime))
121 {}
122
123 void run (Proc& proc) override { (*this) (proc); }
124};
125
126template <typename Proc>
127class CommandFifo final
128{
129public:
130 explicit CommandFifo (int size)
131 : buffer ((size_t) size),
132 abstractFifo (size)
133 {}
134
135 CommandFifo()
136 : CommandFifo (1024)
137 {}
138
139 template <typename Item>
140 void push (Item&& item) noexcept
141 {
142 auto command = makeCommand (std::forward<Item> (item));
143
144 abstractFifo.write (1).forEach ([&] (int index)
145 {
146 buffer[size_t (index)] = std::move (command);
147 });
148 }
149
150 void call (Proc& proc) noexcept
151 {
152 abstractFifo.read (abstractFifo.getNumReady()).forEach ([&] (int index)
153 {
154 buffer[size_t (index)]->run (proc);
155 });
156 }
157
158private:
159 template <typename Func>
160 static std::unique_ptr<Command<Proc>> makeCommand (Func&& func)
161 {
162 using Decayed = std::decay_t<Func>;
163 return std::make_unique<TemplateCommand<Proc, Decayed>> (std::forward<Func> (func));
164 }
165
166 std::vector<std::unique_ptr<Command<Proc>>> buffer;
167 AbstractFifo abstractFifo;
168};

336 template <typename Element>
337 void render (AudioBuffer<Element>& outputBuffer, int startSample, int numSamples)
338 {
339 jassert (samplerSound->getSample() != nullptr);
340
341 auto loopPoints = samplerSound->getLoopPointsInSeconds();
342 loopBegin.setTargetValue (loopPoints.getStart() * samplerSound->getSample()->getSampleRate());
343 loopEnd .setTargetValue (loopPoints.getEnd() * samplerSound->getSample()->getSampleRate());
344
345 auto& data = samplerSound->getSample()->getBuffer();
346
347 auto inL = data.getReadPointer (0);
348 auto inR = data.getNumChannels() > 1 ? data.getReadPointer (1) : nullptr;
349
350 auto outL = outputBuffer.getWritePointer (0, startSample);
351
352 if (outL == nullptr)
353 return;
354
355 auto outR = outputBuffer.getNumChannels() > 1 ? outputBuffer.getWritePointer (1, startSample)
356 : nullptr;
357
358 size_t writePos = 0;
359
360 while (--numSamples >= 0 && renderNextSample (inL, inR, outL, outR, writePos))
361 writePos += 1;
362 }

364 template <typename Element>
365 bool renderNextSample (const float* inL,
366 const float* inR,
367 Element* outL,
368 Element* outR,
369 size_t writePos)
370 {
371 auto currentLevel = level.getNextValue();
372 auto currentFrequency = frequency.getNextValue();
373 auto currentLoopBegin = loopBegin.getNextValue();
374 auto currentLoopEnd = loopEnd.getNextValue();
375
376 if (isTailingOff())
377 {
378 currentLevel *= tailOff;
379 tailOff *= 0.9999;
380
381 if (tailOff < 0.005)
382 {
383 stopNote();
384 return false;
385 }
386 }
387
388 auto pos = (int) currentSamplePos;
389 auto nextPos = pos + 1;
390 auto alpha = (Element) (currentSamplePos - pos);
391 auto invAlpha = 1.0f - alpha;
392
393 // just using a very simple linear interpolation here..
394 auto l = static_cast<Element> (currentLevel * (inL[pos] * invAlpha + inL[nextPos] * alpha));
395 auto r = static_cast<Element> ((inR != nullptr) ? currentLevel * (inR[pos] * invAlpha + inR[nextPos] * alpha)
396 : l);
397
398 if (outR != nullptr)
399 {
400 outL[writePos] += l;
401 outR[writePos] += r;
402 }
403 else
404 {
405 outL[writePos] += (l + r) * 0.5f;
406 }
407
408 std::tie (currentSamplePos, currentDirection) = getNextState (currentFrequency,
409 currentLoopBegin,
410 currentLoopEnd);
411
412 if (currentSamplePos > samplerSound->getSample()->getLength())
413 {
414 stopNote();
415 return false;
416 }
417
418 return true;
419 }
420
421 double getSampleValue() const;
422
423 bool isTailingOff() const
424 {
425 return tailOff != 0.0;
426 }
427
428 void stopNote()
429 {
430 clearCurrentNote();
431 currentSamplePos = 0.0;
432 }

434 enum class Direction
435 {
436 forward,
437 backward
438 };
439
440 std::tuple<double, Direction> getNextState (double freq,
441 double begin,
442 double end) const
443 {
444 auto nextPitchRatio = freq / samplerSound->getCentreFrequencyInHz();
445
446 auto nextSamplePos = currentSamplePos;
447 auto nextDirection = currentDirection;
448
449 // Move the current sample pos in the correct direction
450 switch (currentDirection)
451 {
452 case Direction::forward:
453 nextSamplePos += nextPitchRatio;
454 break;
455
456 case Direction::backward:
457 nextSamplePos -= nextPitchRatio;
458 break;
459
460 default:
461 break;
462 }
463
464 // Update current sample position, taking loop mode into account
465 // If the loop mode was changed while we were travelling backwards, deal
466 // with it gracefully.
467 if (nextDirection == Direction::backward && nextSamplePos < begin)
468 {
469 nextSamplePos = begin;
470 nextDirection = Direction::forward;
471
472 return std::tuple<double, Direction> (nextSamplePos, nextDirection);
473 }
474
475 if (samplerSound->getLoopMode() == LoopMode::none)
476 return std::tuple<double, Direction> (nextSamplePos, nextDirection);
477
478 if (nextDirection == Direction::forward && end < nextSamplePos && !isTailingOff())
479 {
480 if (samplerSound->getLoopMode() == LoopMode::forward)
481 nextSamplePos = begin;
482 else if (samplerSound->getLoopMode() == LoopMode::pingpong)
483 {
484 nextSamplePos = end;
485 nextDirection = Direction::backward;
486 }
487 }
488 return std::tuple<double, Direction> (nextSamplePos, nextDirection);
489 }
490
491 std::shared_ptr<const MPESamplerSound> samplerSound;
492 SmoothedValue<double> level { 0 };
493 SmoothedValue<double> frequency { 0 };
494 SmoothedValue<double> loopBegin;
495 SmoothedValue<double> loopEnd;
496 double previousPressure { 0 };
497 double currentSamplePos { 0 };
498 double tailOff { 0 };
499 Direction currentDirection { Direction::forward };
500 double smoothingLengthInSeconds { 0.01 };
501};
