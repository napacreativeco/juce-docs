52//==============================================================================
53class GainProcessor : public AudioProcessor
54{
55public:
56
57 //==============================================================================
58 GainProcessor()
59 : AudioProcessor (BusesProperties().withInput ("Input", AudioChannelSet::stereo())
60 .withOutput ("Output", AudioChannelSet::stereo()))
61 {
62 addParameter (gain = new AudioParameterFloat ({ "gain", 1 }, "Gain", 0.0f, 1.0f, 0.5f));
63 }
64
65 //==============================================================================
66 void prepareToPlay (double, int) override {}
67 void releaseResources() override {}
68
69 void processBlock (AudioBuffer<float>& buffer, MidiBuffer&) override
70 {
71 buffer.applyGain (*gain);
72 }
73
74 void processBlock (AudioBuffer<double>& buffer, MidiBuffer&) override
75 {
76 buffer.applyGain ((float) *gain);
77 }
78
79 //==============================================================================
80 AudioProcessorEditor* createEditor() override { return new GenericAudioProcessorEditor (*this); }
81 bool hasEditor() const override { return true; }
82
83 //==============================================================================
84 const String getName() const override { return "Gain PlugIn"; }
85 bool acceptsMidi() const override { return false; }
86 bool producesMidi() const override { return false; }
87 double getTailLengthSeconds() const override { return 0; }
88
89 //==============================================================================
90 int getNumPrograms() override { return 1; }
91 int getCurrentProgram() override { return 0; }
92 void setCurrentProgram (int) override {}
93 const String getProgramName (int) override { return "None"; }
94 void changeProgramName (int, const String&) override {}
95
96 //==============================================================================
97 void getStateInformation (MemoryBlock& destData) override
98 {
99 MemoryOutputStream (destData, true).writeFloat (*gain);
100 }
101
102 void setStateInformation (const void* data, int sizeInBytes) override
103 {
104 gain->setValueNotifyingHost (MemoryInputStream (data, static_cast<size_t> (sizeInBytes), false).readFloat());
105 }
106
107 //==============================================================================
108 bool isBusesLayoutSupported (const BusesLayout& layouts) const override
109 {
110 const auto& mainInLayout = layouts.getChannelSet (true, 0);
111 const auto& mainOutLayout = layouts.getChannelSet (false, 0);
112
113 return (mainInLayout == mainOutLayout && (! mainInLayout.isDisabled()));
114 }
115
116private:
117 //==============================================================================
118 AudioParameterFloat* gain;
