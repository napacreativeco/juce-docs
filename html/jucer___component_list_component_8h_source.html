76 struct NamespaceItem : public JucerTreeViewBase
77 {
78 NamespaceItem (const ClassDatabase::Namespace* n)
79 {
80 setNamespace (n);
81 }
82
83 void setNamespace (const ClassDatabase::Namespace* newNamespace)
84 {
85 namespaceToShow = newNamespace;
86 uniqueID = namespaceToShow != nullptr ? "ns_" + namespaceToShow->fullName : "null";
87 refreshSubItems();
88 }
89
90 String getRenamingName() const override { return getDisplayName(); }
91 String getDisplayName() const override { return (namespaceToShow != nullptr ? namespaceToShow->name : String()) + "::"; }
92 void setName (const String&) override {}
93 bool isMissing() const override { return false; }
94 Icon getIcon() const override { return Icon (getIcons().graph, getContentColour (true)); }
95 bool canBeSelected() const override { return true; }
96 bool mightContainSubItems() override { return namespaceToShow != nullptr && ! namespaceToShow->isEmpty(); }
97 String getUniqueName() const override { return uniqueID; }
98
99 void addSubItems() override
100 {
101 if (namespaceToShow != nullptr)
102 {
103 Array<ClassItem*> newComps;
104 Array<NamespaceItem*> newNamespaces;
105
106 for (const auto& c : namespaceToShow->components)
107 newComps.addSorted (*this, new ClassItem (c, *namespaceToShow));
108
109 for(const auto& n : namespaceToShow->namespaces)
110 {
111 if (n.getTotalClassesAndNamespaces() < 10)
112 createFlatItemList (n, newComps);
113 else
114 newNamespaces.add (new NamespaceItem (&n));
115 }
116
117 for (auto c : newComps)
118 addSubItem (c);
119
120 for (auto n : newNamespaces)
121 addSubItem (n);
122 }
123 }
124
125 void createFlatItemList (const ClassDatabase::Namespace& ns, Array<ClassItem*>& newComps)
126 {
127 for (const auto& c : ns.components)
128 newComps.addSorted (*this, new ClassItem (c, *namespaceToShow));
129
130 for (const auto& n : ns.namespaces)
131 createFlatItemList (n, newComps);
132 }

134 static int compareElements (ClassItem* c1, ClassItem* c2)

137 }
