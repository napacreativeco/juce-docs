31class GlobalPathsWindowComponent : public Component,
32 private Timer,

36public:
37 GlobalPathsWindowComponent()
38 {
39 addChildComponent (rescanJUCEPathButton);
40 rescanJUCEPathButton.onClick = [this]
41 {
42 ProjucerApplication::getApp().rescanJUCEPathModules();
43 lastJUCEModulePath = getAppSettings().getStoredPath (Ids::defaultJuceModulePath, TargetOS::getThisOS()).get();
44 };
45
46 addChildComponent (rescanUserPathButton);
47 rescanUserPathButton.onClick = [this]
48 {
49 ProjucerApplication::getApp().rescanUserPathModules();
50 lastUserModulePath = getAppSettings().getStoredPath (Ids::defaultUserModulePath, TargetOS::getThisOS()).get();
51 };
52
53 addChildComponent (warnAboutJUCEPathButton);
54 warnAboutJUCEPathButton.setToggleState (ProjucerApplication::getApp().shouldPromptUserAboutIncorrectJUCEPath(),
55 dontSendNotification);
56 warnAboutJUCEPathButton.onClick = [this]
57 {
58 ProjucerApplication::getApp().setShouldPromptUserAboutIncorrectJUCEPath (warnAboutJUCEPathButton.getToggleState());
59 };
60
61 getGlobalProperties().addChangeListener (this);
62
63 addAndMakeVisible (resetToDefaultsButton);
64 resetToDefaultsButton.onClick = [this] { resetCurrentOSPathsToDefaults(); };
65
66 addAndMakeVisible (propertyViewport);
67 propertyViewport.setViewedComponent (&propertyGroup, false);
68
69 auto os = TargetOS::getThisOS();
70
71 if (os == TargetOS::osx) selectedOSValue = "osx";
72 else if (os == TargetOS::windows) selectedOSValue = "windows";
73 else if (os == TargetOS::linux) selectedOSValue = "linux";
74
75 selectedOSValue.addListener (this);
76
77 buildProps();
78
79 lastJUCEModulePath = getAppSettings().getStoredPath (Ids::defaultJuceModulePath, TargetOS::getThisOS()).get();
80 lastUserModulePath = getAppSettings().getStoredPath (Ids::defaultUserModulePath, TargetOS::getThisOS()).get();
81 }

83 ~GlobalPathsWindowComponent() override
84 {
85 getGlobalProperties().removeChangeListener (this);
86
87 auto juceValue = getAppSettings().getStoredPath (Ids::defaultJuceModulePath, TargetOS::getThisOS());
88 auto userValue = getAppSettings().getStoredPath (Ids::defaultUserModulePath, TargetOS::getThisOS());
89
90 if (juceValue.get() != lastJUCEModulePath) ProjucerApplication::getApp().rescanJUCEPathModules();
91 if (userValue.get() != lastUserModulePath) ProjucerApplication::getApp().rescanUserPathModules();
92 }

123 void highlightJUCEPath()
124 {
125 if (isTimerRunning() || ! isSelectedOSThisOS())
126 return;
127
128 const auto findJucePathPropertyComponent = [this]() -> PropertyComponent*
129 {
130 for (const auto& prop : propertyGroup.getProperties())
131 if (prop->getName() == "Path to JUCE")
132 return prop.get();
133
134 return nullptr;
135 };
136
137 if (auto* propComponent = findJucePathPropertyComponent())
138 {
139 boundsToHighlight = getLocalArea (nullptr, propComponent->getScreenBounds());
140 flashAlpha = 0.0f;
141 hasFlashed = false;
142
143 startTimer (25);
144 }
145 }

147private:
148 //==============================================================================
149 void timerCallback() override
150 {
151 flashAlpha += (hasFlashed ? -0.05f : 0.05f);
152
153 if (flashAlpha > 0.75f)
154 {
155 hasFlashed = true;
156 }
157 else if (flashAlpha < 0.0f)
158 {
159 flashAlpha = 0.0f;
160 boundsToHighlight = {};
161
162 stopTimer();
163 }
164
165 repaint();
166 }

181 bool isSelectedOSThisOS() { return TargetOS::getThisOS() == getSelectedOS(); }

183 TargetOS::OS getSelectedOS() const
184 {
185 auto val = selectedOSValue.getValue();
186
187 if (val == "osx") return TargetOS::osx;
188 else if (val == "windows") return TargetOS::windows;
189 else if (val == "linux") return TargetOS::linux;
190
191 jassertfalse;
192 return TargetOS::unknown;
193 }

263 void updateValues()
264 {

267
268 jucePathValue = settings.getStoredPath (Ids::jucePath, os);
269 juceModulePathValue = settings.getStoredPath (Ids::defaultJuceModulePath, os);
270 userModulePathValue = settings.getStoredPath (Ids::defaultUserModulePath, os);
271 vstPathValue = settings.getStoredPath (Ids::vstLegacyPath, os);
272 aaxPathValue = settings.getStoredPath (Ids::aaxPath, os);
273 araPathValue = settings.getStoredPath (Ids::araPath, os);
