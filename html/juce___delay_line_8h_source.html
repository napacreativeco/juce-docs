171 template <typename ProcessContext>
172 void process (const ProcessContext& context) noexcept
173 {
174 const auto& inputBlock = context.getInputBlock();
175 auto& outputBlock = context.getOutputBlock();
176 const auto numChannels = outputBlock.getNumChannels();
177 const auto numSamples = outputBlock.getNumSamples();
178
179 jassert (inputBlock.getNumChannels() == numChannels);
180 jassert (inputBlock.getNumChannels() == writePos.size());
181 jassert (inputBlock.getNumSamples() == numSamples);
182
183 if (context.isBypassed)
184 {
185 outputBlock.copyFrom (inputBlock);
186 return;
187 }
188
189 for (size_t channel = 0; channel < numChannels; ++channel)
190 {
191 auto* inputSamples = inputBlock.getChannelPointer (channel);
192 auto* outputSamples = outputBlock.getChannelPointer (channel);
193
194 for (size_t i = 0; i < numSamples; ++i)
195 {
196 pushSample ((int) channel, inputSamples[i]);
197 outputSamples[i] = popSample ((int) channel);
198 }
199 }
200 }

203 //==============================================================================
204 SampleType interpolateSample (int channel)
205 {
206 if constexpr (std::is_same_v<InterpolationType, DelayLineInterpolationTypes::None>)
207 {
208 auto index = (readPos[(size_t) channel] + delayInt) % totalSize;
209 return bufferData.getSample (channel, index);
210 }
211 else if constexpr (std::is_same_v<InterpolationType, DelayLineInterpolationTypes::Linear>)
212 {
213 auto index1 = readPos[(size_t) channel] + delayInt;
214 auto index2 = index1 + 1;
215
216 if (index2 >= totalSize)
217 {
218 index1 %= totalSize;
219 index2 %= totalSize;
220 }
221
222 auto value1 = bufferData.getSample (channel, index1);
223 auto value2 = bufferData.getSample (channel, index2);
224
225 return value1 + delayFrac * (value2 - value1);
226 }
227 else if constexpr (std::is_same_v<InterpolationType, DelayLineInterpolationTypes::Lagrange3rd>)
228 {
229 auto index1 = readPos[(size_t) channel] + delayInt;
230 auto index2 = index1 + 1;
231 auto index3 = index2 + 1;
232 auto index4 = index3 + 1;
233
234 if (index4 >= totalSize)
235 {
236 index1 %= totalSize;
237 index2 %= totalSize;
238 index3 %= totalSize;
239 index4 %= totalSize;
240 }
241
242 auto* samples = bufferData.getReadPointer (channel);
243
244 auto value1 = samples[index1];
245 auto value2 = samples[index2];
246 auto value3 = samples[index3];
247 auto value4 = samples[index4];
248
249 auto d1 = delayFrac - 1.f;
250 auto d2 = delayFrac - 2.f;
251 auto d3 = delayFrac - 3.f;
252
253 auto c1 = -d1 * d2 * d3 / 6.f;
254 auto c2 = d2 * d3 * 0.5f;
255 auto c3 = -d1 * d3 * 0.5f;
256 auto c4 = d1 * d2 / 6.f;
257
258 return value1 * c1 + delayFrac * (value2 * c2 + value3 * c3 + value4 * c4);
259 }
260 else if constexpr (std::is_same_v<InterpolationType, DelayLineInterpolationTypes::Thiran>)
261 {
262 auto index1 = readPos[(size_t) channel] + delayInt;
263 auto index2 = index1 + 1;
264
265 if (index2 >= totalSize)
266 {
267 index1 %= totalSize;
268 index2 %= totalSize;
269 }
270
271 auto value1 = bufferData.getSample (channel, index1);
272 auto value2 = bufferData.getSample (channel, index2);
273
274 auto output = delayFrac == 0 ? value1 : value2 + alpha * (value1 - v[(size_t) channel]);
275 v[(size_t) channel] = output;
276
277 return output;
278 }
279 }

282 void updateInternalVariables()
283 {
284 if constexpr (std::is_same_v<InterpolationType, DelayLineInterpolationTypes::Lagrange3rd>)
285 {
286 if (delayInt >= 1)
287 {
288 delayFrac++;
289 delayInt--;
290 }
291 }
292 else if constexpr (std::is_same_v<InterpolationType, DelayLineInterpolationTypes::Thiran>)
293 {
294 if (delayFrac < (SampleType) 0.618 && delayInt >= 1)
295 {
296 delayFrac++;
297 delayInt--;
298 }
299
300 alpha = (1 - delayFrac) / (1 + delayFrac);
301 }
302 }
