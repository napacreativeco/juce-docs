72class JUCE_API AccessibilityActions

74public:

80
88 AccessibilityActions& addAction (AccessibilityActionType type,
89 std::function<void()> actionCallback)
90 {
91 actionMap[type] = std::move (actionCallback);
92 return *this;
93 }
94
96 bool contains (AccessibilityActionType type) const
97 {
98 return actionMap.find (type) != actionMap.end();
99 }
100
104 bool invoke (AccessibilityActionType type) const
105 {
106 auto iter = actionMap.find (type);
107
108 if (iter == actionMap.end())
109 return false;
110
111 iter->second();
112 return true;
113 }
114
115private:
116 std::map<AccessibilityActionType, std::function<void()>> actionMap;
117};
