31template <class ComponentType>
32class ComponentUndoableAction : public UndoableAction
33{
34public:
35 ComponentUndoableAction (ComponentType* const comp,
36 ComponentLayout& layout_)
37 : layout (layout_),
38 componentIndex (layout_.indexOfComponent (comp))
39 {
40 jassert (comp != nullptr);
41 jassert (componentIndex >= 0);
42 }
43
44 ComponentType* getComponent() const
45 {
46 ComponentType* const c = dynamic_cast<ComponentType*> (layout.getComponent (componentIndex));
47 jassert (c != nullptr);
48 return c;
49 }
50
51 int getSizeInUnits() { return 2; }
52
53protected:
54 ComponentLayout& layout;
55 const int componentIndex;
56
57 void changed() const
58 {
59 jassert (layout.getDocument() != nullptr);
60 layout.getDocument()->changed();
61 }
62
63 void showCorrectTab() const
64 {
65 if (JucerDocumentEditor* const ed = JucerDocumentEditor::getActiveDocumentHolder())
66 ed->showLayout();
67
68 if (layout.getSelectedSet().getNumSelected() == 0)
69 if (ComponentType* const c = dynamic_cast<ComponentType*> (layout.getComponent (componentIndex)))
70 layout.getSelectedSet().selectOnly (getComponent());
71 }
72
73private:
74 JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR (ComponentUndoableAction)
75};
