42 static const ProjectType* findType (const String& typeCode)
43 {
44 const auto& types = getAllTypes();
45
46 for (auto i = types.size(); --i >= 0;)

132 if (name == "Static Library") return Type::StaticLibrary;
133 if (name == "Dynamic Library") return Type::DynamicLibrary;
134 if (name == "VST") return Type::VSTPlugIn;
135 if (name == "VST3") return Type::VST3PlugIn;
