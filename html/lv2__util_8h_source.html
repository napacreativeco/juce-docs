40static inline void*
41lv2_features_data(const LV2_Feature* const* features, const char* const uri)
42{
43 if (features) {
44 for (const LV2_Feature* const* f = features; *f; ++f) {
45 if (!strcmp(uri, (*f)->URI)) {
46 return (*f)->data;
47 }
48 }
49 }
50 return NULL;
51}

75static inline const char*
76lv2_features_query(const LV2_Feature* const* features, ...)
77{
78 va_list args;
79 va_start(args, features);
80
81 const char* uri = NULL;
82 while ((uri = va_arg(args, const char*))) {
83 void** data = va_arg(args, void**);
84 bool required = va_arg(args, int);
85
86 *data = lv2_features_data(features, uri);
87 if (required && !*data) {
88 va_end(args);
89 return uri;
90 }
91 }
92
93 va_end(args);
94 return NULL;
95}
