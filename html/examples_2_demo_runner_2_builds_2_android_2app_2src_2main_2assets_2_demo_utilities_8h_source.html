48inline File getExamplesDirectory() noexcept
49{
50 #ifdef PIP_JUCE_EXAMPLES_DIRECTORY
51 MemoryOutputStream mo;
52
53 auto success = Base64::convertFromBase64 (mo, JUCE_STRINGIFY (PIP_JUCE_EXAMPLES_DIRECTORY));
54 ignoreUnused (success);
55 jassert (success);
56
57 return mo.toString();
58 #elif defined PIP_JUCE_EXAMPLES_DIRECTORY_STRING
59 return File { CharPointer_UTF8 { PIP_JUCE_EXAMPLES_DIRECTORY_STRING } };
60 #else
61 auto currentFile = File::getSpecialLocation (File::SpecialLocationType::currentApplicationFile);
62 auto exampleDir = currentFile.getParentDirectory().getChildFile ("examples");
63
64 if (exampleDir.exists())
65 return exampleDir;
66
67 // keep track of the number of parent directories so we don't go on endlessly
68 for (int numTries = 0; numTries < 15; ++numTries)
69 {
70 if (currentFile.getFileName() == "examples")
71 return currentFile;
72
73 const auto sibling = currentFile.getSiblingFile ("examples");
74
75 if (sibling.exists())
76 return sibling;
77
78 currentFile = currentFile.getParentDirectory();
79 }
80
81 return currentFile;
82 #endif
83}

85inline std::unique_ptr<InputStream> createAssetInputStream (const char* resourcePath)
86{
87 #if JUCE_ANDROID
88 ZipFile apkZip (File::getSpecialLocation (File::invokedExecutableFile));
89 return std::unique_ptr<InputStream> (apkZip.createStreamForEntry (apkZip.getIndexOfFileName ("assets/" + String (resourcePath))));
90 #else
91 #if JUCE_IOS
92 auto assetsDir = File::getSpecialLocation (File::currentExecutableFile)
93 .getParentDirectory().getChildFile ("Assets");
94 #elif JUCE_MAC
95 auto assetsDir = File::getSpecialLocation (File::currentExecutableFile)
96 .getParentDirectory().getParentDirectory().getChildFile ("Resources").getChildFile ("Assets");
97
98 if (! assetsDir.exists())
99 assetsDir = getExamplesDirectory().getChildFile ("Assets");
100 #else
101 auto assetsDir = getExamplesDirectory().getChildFile ("Assets");
102 #endif
103
104 auto resourceFile = assetsDir.getChildFile (resourcePath);
105 jassert (resourceFile.existsAsFile());
106
107 return resourceFile.createInputStream();
108 #endif
109}

111inline Image getImageFromAssets (const char* assetName)
112{
113 auto hashCode = (String (assetName) + "@juce_demo_assets").hashCode64();
114 auto img = ImageCache::getFromHashCode (hashCode);
115
116 if (img.isNull())
117 {
118 std::unique_ptr<InputStream> juceIconStream (createAssetInputStream (assetName));
119
120 if (juceIconStream == nullptr)
121 return {};
122
123 img = ImageFileFormat::loadFrom (*juceIconStream);
124
125 ImageCache::addImageToCache (img, hashCode);
126 }
127
128 return img;
129}

217// This is basically a sawtooth wave generator - maps a value that bounces between
218// 0.0 and 1.0 at a random speed
219struct BouncingNumber
220{
221 BouncingNumber()
222 : speed (0.0004 + 0.0007 * Random::getSystemRandom().nextDouble()),
223 phase (Random::getSystemRandom().nextDouble())
224 {
225 }
226
227 float getValue() const
228 {
229 double v = fmod (phase + speed * Time::getMillisecondCounterHiRes(), 2.0);
230 return (float) (v >= 1.0 ? (2.0 - v) : v);
231 }
232
233protected:
234 double speed, phase;
235};
236
237struct SlowerBouncingNumber : public BouncingNumber
238{
239 SlowerBouncingNumber()
240 {
241 speed *= 0.3;
242 }
243};
