466class MPEDemoSynthVoice : public MPESynthesiserVoice
467{
468public:
469 //==============================================================================
470 MPEDemoSynthVoice() {}
471
472 //==============================================================================
473 void noteStarted() override
474 {
475 jassert (currentlyPlayingNote.isValid());
476 jassert (currentlyPlayingNote.keyState == MPENote::keyDown
477 || currentlyPlayingNote.keyState == MPENote::keyDownAndSustained);
478
479 level .setTargetValue (currentlyPlayingNote.pressure.asUnsignedFloat());
480 frequency.setTargetValue (currentlyPlayingNote.getFrequencyInHertz());
481 timbre .setTargetValue (currentlyPlayingNote.timbre.asUnsignedFloat());
482
483 phase = 0.0;
484 auto cyclesPerSample = frequency.getNextValue() / currentSampleRate;
485 phaseDelta = MathConstants<double>::twoPi * cyclesPerSample;
486
487 tailOff = 0.0;
488 }
489
490 void noteStopped (bool allowTailOff) override
491 {
492 jassert (currentlyPlayingNote.keyState == MPENote::off);
493
494 if (allowTailOff)
495 {
496 // start a tail-off by setting this flag. The render callback will pick up on
497 // this and do a fade out, calling clearCurrentNote() when it's finished.
498
499 if (tailOff == 0.0) // we only need to begin a tail-off if it's not already doing so - the
500 // stopNote method could be called more than once.
501 tailOff = 1.0;
502 }
503 else
504 {
505 // we're being told to stop playing immediately, so reset everything..
506 clearCurrentNote();
507 phaseDelta = 0.0;
508 }
509 }
510
511 void notePressureChanged() override
512 {
513 level.setTargetValue (currentlyPlayingNote.pressure.asUnsignedFloat());
514 }
515
516 void notePitchbendChanged() override
517 {
518 frequency.setTargetValue (currentlyPlayingNote.getFrequencyInHertz());
519 }
520
521 void noteTimbreChanged() override
522 {
523 timbre.setTargetValue (currentlyPlayingNote.timbre.asUnsignedFloat());
524 }
525
526 void noteKeyStateChanged() override {}
527
528 void setCurrentSampleRate (double newRate) override
529 {
530 if (currentSampleRate != newRate)
531 {
532 noteStopped (false);
533 currentSampleRate = newRate;
534
535 level .reset (currentSampleRate, smoothingLengthInSeconds);
536 timbre .reset (currentSampleRate, smoothingLengthInSeconds);
537 frequency.reset (currentSampleRate, smoothingLengthInSeconds);
538 }
539 }
540
541 //==============================================================================
542 virtual void renderNextBlock (AudioBuffer<float>& outputBuffer,
543 int startSample,
544 int numSamples) override
545 {
546 if (phaseDelta != 0.0)
547 {
548 if (tailOff > 0.0)
549 {
550 while (--numSamples >= 0)
551 {
552 auto currentSample = getNextSample() * (float) tailOff;
553
554 for (auto i = outputBuffer.getNumChannels(); --i >= 0;)
555 outputBuffer.addSample (i, startSample, currentSample);
556
557 ++startSample;
558
559 tailOff *= 0.99;
560
561 if (tailOff <= 0.005)
562 {
563 clearCurrentNote();
564
565 phaseDelta = 0.0;
566 break;
567 }
568 }
569 }
570 else
571 {
572 while (--numSamples >= 0)
573 {
574 auto currentSample = getNextSample();
575
576 for (auto i = outputBuffer.getNumChannels(); --i >= 0;)
577 outputBuffer.addSample (i, startSample, currentSample);
578
579 ++startSample;
580 }
581 }
582 }
583 }
584
585 using MPESynthesiserVoice::renderNextBlock;
586
587private:
588 //==============================================================================
589 float getNextSample() noexcept
590 {
591 auto levelDb = (level.getNextValue() - 1.0) * maxLevelDb;
592 auto amplitude = pow (10.0f, 0.05f * levelDb) * maxLevel;
593
594 // timbre is used to blend between a sine and a square.
595 auto f1 = std::sin (phase);
596 auto f2 = copysign (1.0, f1);
597 auto a2 = timbre.getNextValue();
598 auto a1 = 1.0 - a2;
599
600 auto nextSample = float (amplitude * ((a1 * f1) + (a2 * f2)));
601
602 auto cyclesPerSample = frequency.getNextValue() / currentSampleRate;
603 phaseDelta = MathConstants<double>::twoPi * cyclesPerSample;
604 phase = std::fmod (phase + phaseDelta, MathConstants<double>::twoPi);
605
606 return nextSample;
607 }
608
609 //==============================================================================
610 SmoothedValue<double> level, timbre, frequency;
611
612 double phase = 0.0;
613 double phaseDelta = 0.0;
614 double tailOff = 0.0;
615
616 const double maxLevel = 0.05;
617 const double maxLevelDb = 31.0;
618 const double smoothingLengthInSeconds = 0.01;
619};
