38 static void scanFileForTranslations (StringArray& strings, const File& file)
39 {
40 auto content = file.loadFileAsString();
41 auto p = content.getCharPointer();
42
43 for (;;)
44 {
45 p = CharacterFunctions::find (p, CharPointer_ASCII ("TRANS"));
46
47 if (p.isEmpty())
48 break;
49
50 p += 5;
51 p.incrementToEndOfWhitespace();
52
53 if (*p == '(')
54 {
55 ++p;
56 MemoryOutputStream text;
57 parseStringLiteral (p, text);
58
59 addString (strings, text.toString());
60 }
61 }
62 }

64 static void parseStringLiteral (String::CharPointerType& p, MemoryOutputStream& out) noexcept
65 {
66 p.incrementToEndOfWhitespace();
67
68 if (p.getAndAdvance() == '"')
69 {
70 auto start = p;
71
72 for (;;)
73 {
74 auto c = *p;
75
76 if (c == '"')
77 {
78 out << String (start, p);
79 ++p;
80 parseStringLiteral (p, out);
81 return;
82 }
83
84 if (c == 0)
85 break;
86
87 if (c == '\\')
88 {
89 out << String (start, p);
90 ++p;
91 out << String::charToString (readEscapedChar (p));
92 start = p + 1;
93 }
94
95 ++p;
96 }
97 }
98 }

100 static juce_wchar readEscapedChar (String::CharPointerType& p)
101 {
102 auto c = *p;
103
104 switch (c)
105 {
106 case '"':
107 case '\\':
108 case '/': break;
109
110 case 'b': c = '\b'; break;
111 case 'f': c = '\f'; break;
112 case 'n': c = '\n'; break;
113 case 'r': c = '\r'; break;
114 case 't': c = '\t'; break;
115
116 case 'x':
117 ++p;
118 c = 0;
119
120 for (int i = 4; --i >= 0;)
121 {
122 const int digitValue = CharacterFunctions::getHexDigitValue (*p);
123 if (digitValue < 0)
124 break;
125
126 ++p;
127 c = (c << 4) + (juce_wchar) digitValue;
128 }
129
130 break;
131
132 case '0': case '1': case '2': case '3': case '4':
133 case '5': case '6': case '7': case '8': case '9':
134 c = 0;
135
136 for (int i = 4; --i >= 0;)
137 {
138 const auto digitValue = (int) (*p - '0');
139 if (digitValue < 0 || digitValue > 7)
140 break;
141
142 ++p;
143 c = (c << 3) + (juce_wchar) digitValue;
144 }
145
146 break;
147
148 default:
149 break;
150 }
151
152 return c;
153 }

155 static void scanFilesForTranslations (StringArray& strings, const Project::Item& p)
156 {
157 if (p.isFile())
158 {
159 const File file (p.getFile());
160
161 if (file.hasFileExtension (sourceOrHeaderFileExtensions))
162 scanFileForTranslations (strings, file);
163 }
164
165 for (int i = 0; i < p.getNumChildren(); ++i)
166 scanFilesForTranslations (strings, p.getChild (i));
167 }

180 static void scanProject (StringArray& strings, Project& project)
181 {
182 scanFilesForTranslations (strings, project.getMainGroup());
183
184 OwnedArray<LibraryModule> modules;
185 project.getEnabledModules().createRequiredModules (modules);
186
187 for (int j = 0; j < modules.size(); ++j)
188 {
189 const File localFolder (modules.getUnchecked(j)->getFolder());
190
191 Array<File> files;
192 modules.getUnchecked(j)->findBrowseableFiles (localFolder, files);
193
194 for (int i = 0; i < files.size(); ++i)
195 scanFileForTranslations (strings, files.getReference(i));
196 }

201 static StringArray breakApart (const String& munged)
202 {
203 StringArray lines, result;
204 lines.addLines (munged);
205
206 String currentItem;
207
208 for (int i = 0; i < lines.size(); ++i)
209 {
210 if (lines[i].contains (getMungingSeparator()))
211 {
212 if (currentItem.isNotEmpty())
213 result.add (currentItem);
214
215 currentItem = String();
216 }
217 else
218 {
219 if (currentItem.isNotEmpty())
220 currentItem << newLine;
221
222 currentItem << lines[i];
223 }
224 }
225
226 if (currentItem.isNotEmpty())
227 result.add (currentItem);
228
229 return result;
230 }

261 static String mungeStrings (const StringArray& strings)
262 {
263 MemoryOutputStream s;
264
265 for (int i = 0; i < strings.size(); ++i)
266 {
267 s << getMungingSeparator() << i << "." << newLine << strings[i];
268
269 if (i < strings.size() - 1)
270 s << newLine;
271 }
272
273 return s.toString();
274 }

283 static String createFinishedTranslationFile (StringArray preStrings,
284 StringArray postStrings,
285 const LocalisedStrings& original)
286 {
287 const StringPairArray& originalStrings (original.getMappings());
288
289 StringArray lines;
290
291 if (originalStrings.size() > 0)
292 {
293 lines.add ("language: " + original.getLanguageName());
294 lines.add ("countries: " + original.getCountryCodes().joinIntoString (" "));
295 lines.add (String());
296
297 const StringArray& originalKeys (originalStrings.getAllKeys());
298 const StringArray& originalValues (originalStrings.getAllValues());
299
300 for (int i = preStrings.size(); --i >= 0;)
301 {
302 if (originalKeys.contains (preStrings[i]))
303 {
304 preStrings.remove (i);
305 postStrings.remove (i);
306 }
307 }
308
309 for (int i = 0; i < originalStrings.size(); ++i)
310 lines.add (createLine (originalKeys[i], originalValues[i]));
311 }
312 else
313 {
314 lines.add ("language: [enter full name of the language here!]");
315 lines.add ("countries: [enter list of 2-character country codes here!]");
316 lines.add (String());
317 }
318
319 for (int i = 0; i < preStrings.size(); ++i)
320 lines.add (createLine (preStrings[i], postStrings[i]));
321
322 return lines.joinIntoString (newLine);
323 }
