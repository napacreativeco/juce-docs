100 static Matrix3D fromFrustum (Type left, Type right, Type bottom, Type top, Type nearDistance, Type farDistance) noexcept
101 {
102 return { (Type (2) * nearDistance) / (right - left), 0, 0, 0,
103 0, (Type (2) * nearDistance) / (top - bottom), 0, 0,
104 (right + left) / (right - left), (top + bottom) / (top - bottom), -(farDistance + nearDistance) / (farDistance - nearDistance), Type (-1),
105 0, 0, -(Type (2) * farDistance * nearDistance) / (farDistance - nearDistance), 0 };
106 }

109 static Matrix3D rotation (Vector3D<Type> eulerAngleRadians) noexcept
110 {
111 auto cx = std::cos (eulerAngleRadians.x), sx = std::sin (eulerAngleRadians.x),
112 cy = std::cos (eulerAngleRadians.y), sy = std::sin (eulerAngleRadians.y),
113 cz = std::cos (eulerAngleRadians.z), sz = std::sin (eulerAngleRadians.z);
114
115 return { (cy * cz) + (sx * sy * sz), cx * sz, (cy * sx * sz) - (cz * sy), 0,
116 (cz * sx * sy) - (cy * sz), cx * cz, (cy * cz * sx) + (sy * sz), 0,
117 cx * sy, -sx, cx * cy, 0,
118 0, 0, 0, Type (1) };
119 }
