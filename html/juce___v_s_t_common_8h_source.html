42 bool matches (const Array<ChannelType>& chans) const noexcept
43 {
44 auto n = static_cast<int> (sizeof (channels) / sizeof (ChannelType));
45
46 for (int i = 0; i < n; ++i)
47 {
48 if (channels[i] == unknown) return (i == chans.size());
49 if (i == chans.size()) return (channels[i] == unknown);
50
51 if (channels[i] != chans.getUnchecked (i))
52 return false;
53 }
54
55 return true;
56 }
57 };

59 static AudioChannelSet vstArrangementTypeToChannelSet (int32 arr, int fallbackNumChannels)
60 {
61 if (arr == Vst2::kSpeakerArrEmpty) return AudioChannelSet::disabled();
62 else if (arr == Vst2::kSpeakerArrMono) return AudioChannelSet::mono();
63 else if (arr == Vst2::kSpeakerArrStereo) return AudioChannelSet::stereo();
64 else if (arr == Vst2::kSpeakerArr30Cine) return AudioChannelSet::createLCR();
65 else if (arr == Vst2::kSpeakerArr30Music) return AudioChannelSet::createLRS();
66 else if (arr == Vst2::kSpeakerArr40Cine) return AudioChannelSet::createLCRS();
67 else if (arr == Vst2::kSpeakerArr50) return AudioChannelSet::create5point0();
68 else if (arr == Vst2::kSpeakerArr51) return AudioChannelSet::create5point1();
69 else if (arr == Vst2::kSpeakerArr60Cine) return AudioChannelSet::create6point0();
70 else if (arr == Vst2::kSpeakerArr61Cine) return AudioChannelSet::create6point1();
71 else if (arr == Vst2::kSpeakerArr60Music) return AudioChannelSet::create6point0Music();
72 else if (arr == Vst2::kSpeakerArr61Music) return AudioChannelSet::create6point1Music();
73 else if (arr == Vst2::kSpeakerArr70Music) return AudioChannelSet::create7point0();
74 else if (arr == Vst2::kSpeakerArr70Cine) return AudioChannelSet::create7point0SDDS();
75 else if (arr == Vst2::kSpeakerArr71Music) return AudioChannelSet::create7point1();
76 else if (arr == Vst2::kSpeakerArr71Cine) return AudioChannelSet::create7point1SDDS();
77 else if (arr == Vst2::kSpeakerArr40Music) return AudioChannelSet::quadraphonic();
78
79 for (const Mapping* m = getMappings(); m->vst2 != Vst2::kSpeakerArrEmpty; ++m)
80 {
81 if (m->vst2 == arr)
82 {
83 AudioChannelSet s;
84
85 for (int i = 0; m->channels[i] != 0; ++i)
86 s.addChannel (m->channels[i]);
87
88 return s;
89 }
90 }
91
92 return AudioChannelSet::discreteChannels (fallbackNumChannels);
93 }

100 static int32 channelSetToVstArrangementType (AudioChannelSet channels)
101 {
102 if (channels == AudioChannelSet::disabled()) return Vst2::kSpeakerArrEmpty;
103 else if (channels == AudioChannelSet::mono()) return Vst2::kSpeakerArrMono;
104 else if (channels == AudioChannelSet::stereo()) return Vst2::kSpeakerArrStereo;
105 else if (channels == AudioChannelSet::createLCR()) return Vst2::kSpeakerArr30Cine;
106 else if (channels == AudioChannelSet::createLRS()) return Vst2::kSpeakerArr30Music;
107 else if (channels == AudioChannelSet::createLCRS()) return Vst2::kSpeakerArr40Cine;
108 else if (channels == AudioChannelSet::create5point0()) return Vst2::kSpeakerArr50;
109 else if (channels == AudioChannelSet::create5point1()) return Vst2::kSpeakerArr51;
110 else if (channels == AudioChannelSet::create6point0()) return Vst2::kSpeakerArr60Cine;
111 else if (channels == AudioChannelSet::create6point1()) return Vst2::kSpeakerArr61Cine;
112 else if (channels == AudioChannelSet::create6point0Music()) return Vst2::kSpeakerArr60Music;
113 else if (channels == AudioChannelSet::create6point1Music()) return Vst2::kSpeakerArr61Music;
114 else if (channels == AudioChannelSet::create7point0()) return Vst2::kSpeakerArr70Music;
115 else if (channels == AudioChannelSet::create7point0SDDS()) return Vst2::kSpeakerArr70Cine;
116 else if (channels == AudioChannelSet::create7point1()) return Vst2::kSpeakerArr71Music;
117 else if (channels == AudioChannelSet::create7point1SDDS()) return Vst2::kSpeakerArr71Cine;
118 else if (channels == AudioChannelSet::quadraphonic()) return Vst2::kSpeakerArr40Music;
119
120 if (channels == AudioChannelSet::disabled())
121 return Vst2::kSpeakerArrEmpty;
122
123 auto chans = channels.getChannelTypes();
124
125 for (auto* m = getMappings(); m->vst2 != Vst2::kSpeakerArrEmpty; ++m)
126 if (m->matches (chans))
127 return m->vst2;
128
129 return Vst2::kSpeakerArrUserDefined;
130 }

132 static void channelSetToVstArrangement (const AudioChannelSet& channels, Vst2::VstSpeakerArrangement& result)
133 {
134 result.type = channelSetToVstArrangementType (channels);
135 result.numChannels = channels.size();
136
137 for (int i = 0; i < result.numChannels; ++i)
138 {
139 auto& speaker = result.speakers[i];
140
141 zeromem (&speaker, sizeof (Vst2::VstSpeakerProperties));
142 speaker.type = getSpeakerType (channels.getTypeOfChannel (i));
143 }
144 }
