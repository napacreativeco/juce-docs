110 NativeContext (Component& comp,
111 const OpenGLPixelFormat& pixelFormat,
112 void* /*contextToShareWith*/,
113 bool useMultisamplingIn,
114 OpenGLVersion)
115 : component (comp)
116 {
117 auto env = getEnv();
118
119 // Do we have a native peer that we can attach to?
120 if (component.getPeer()->getNativeHandle() == nullptr)
121 return;
122
123 // Initialise the EGL display
124 if (! initEGLDisplay (pixelFormat, useMultisamplingIn))
125 return;
126
127 // create a native surface view
128 surfaceView = GlobalRef (LocalRef<jobject>(env->NewObject (JuceOpenGLViewSurface,
129 JuceOpenGLViewSurface.constructor,
130 getAppContext().get(),
131 reinterpret_cast<jlong> (this))));
132 if (surfaceView.get() == nullptr)
133 return;
134
135 // add the view to the view hierarchy
136 // after this the nativecontext can receive callbacks
137 env->CallVoidMethod ((jobject) component.getPeer()->getNativeHandle(),
138 AndroidViewGroup.addView, surfaceView.get());
139
140 // initialise the geometry of the view
141 auto bounds = component.getTopLevelComponent()->getLocalArea (&component, component.getLocalBounds());
142 bounds *= component.getDesktopScaleFactor();
143
144 updateWindowPosition (bounds);
145 hasInitialised = true;
146 }

313 bool initEGLDisplay (const OpenGLPixelFormat& pixelFormat, bool multisample)
314 {
315 // already initialised?
316 if (display != EGL_NO_DISPLAY)
317 return true;
318
319 if ((display = eglGetDisplay (EGL_DEFAULT_DISPLAY)) == EGL_NO_DISPLAY)
320 {
321 jassertfalse;
322 return false;
323 }
324
325 if (! eglInitialize (display, nullptr, nullptr))
326 {
327 jassertfalse;
328 return false;
329 }
330
331 if (tryChooseConfig ({ EGL_SAMPLE_BUFFERS, multisample ? 1 : 0, EGL_SAMPLES, pixelFormat.multisamplingLevel }))
332 return true;
333
334 if (tryChooseConfig ({}))
335 return true;
336
337 eglTerminate (display);
338 jassertfalse;
339 return false;
340 }

348 {
349 auto* env = getEnv();
350
351 const LocalRef<jobject> holder (env->CallObjectMethod (surfaceView.get(), JuceOpenGLViewSurface.getHolder));
352
353 if (holder == nullptr)
354 return nullptr;
355
356 const LocalRef<jobject> jSurface (env->CallObjectMethod (holder.get(), AndroidSurfaceHolder.getSurface));
357
358 if (jSurface == nullptr)
359 return nullptr;

361 constexpr auto numAttempts = 2;
362
363 for (auto i = 0; i < numAttempts; Thread::sleep (200), ++i)
364 if (auto* ptr = ANativeWindow_fromSurface (env, jSurface.get()))
365 return std::unique_ptr<ANativeWindow, NativeWindowReleaser> { ptr };
366
367 return nullptr;
368 }
