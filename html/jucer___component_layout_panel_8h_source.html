68 private ChangeListener
69 {
70 public:
71 LayoutPropsPanel (JucerDocument& doc, ComponentLayout& l)
72 : document (doc), layout (l)
73 {
74 layout.getSelectedSet().addChangeListener (this);
75 addAndMakeVisible (propsPanel);
76 }
77
78 ~LayoutPropsPanel() override
79 {
80 layout.getSelectedSet().removeChangeListener (this);

88
89 void clear()

91 propsPanel.clear();
92 }
93
94 void updateList()
95 {
96 clear();
97
98 auto numSelected = layout.getSelectedSet().getNumSelected();
99
100 if (numSelected > 0) // xxx need to cope with multiple
101 {
102 if (auto* comp = layout.getSelectedSet().getSelectedItem (0))
103 if (auto* type = ComponentTypeHandler::getHandlerFor (*comp))
104 type->addPropertiesToPropertyPanel (comp, document, propsPanel, numSelected > 1);
105 }
106 }
107
108 private:
109 void changeListenerCallback (ChangeBroadcaster*) override
110 {
111 updateList();
112 }

114 JucerDocument& document;
115 ComponentLayout& layout;
116 PropertyPanel propsPanel;
