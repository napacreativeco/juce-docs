24/* This component scrolls a continuous waveform showing the audio that's
25 coming into whatever audio inputs this object is connected to.
26*/
27class LiveScrollingAudioDisplay : public AudioVisualiserComponent,
28 public AudioIODeviceCallback
29{
30public:
31 LiveScrollingAudioDisplay() : AudioVisualiserComponent (1)
32 {
33 setSamplesPerBlock (256);
34 setBufferSize (1024);
35 }
36
37 //==============================================================================
38 void audioDeviceAboutToStart (AudioIODevice*) override
39 {
40 clear();
41 }
42
43 void audioDeviceStopped() override
44 {
45 clear();
46 }
47
48 void audioDeviceIOCallbackWithContext (const float* const* inputChannelData, int numInputChannels,
49 float* const* outputChannelData, int numOutputChannels,
50 int numberOfSamples, const AudioIODeviceCallbackContext& context) override
51 {
52 ignoreUnused (context);
53
54 for (int i = 0; i < numberOfSamples; ++i)
55 {
56 float inputSample = 0;
57
58 for (int chan = 0; chan < numInputChannels; ++chan)
59 if (const float* inputChannel = inputChannelData[chan])
60 inputSample += inputChannel[i]; // find the sum of all the channels
61
62 inputSample *= 10.0f; // boost the level to make it more easily visible.
63
64 pushSample (&inputSample, 1);
65 }
66
67 // We need to clear the output buffers before returning, in case they're full of junk..
68 for (int j = 0; j < numOutputChannels; ++j)
69 if (float* outputChannel = outputChannelData[j])
70 zeromem (outputChannel, (size_t) numberOfSamples * sizeof (float));
71 }
72
73 JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR (LiveScrollingAudioDisplay)
74};
