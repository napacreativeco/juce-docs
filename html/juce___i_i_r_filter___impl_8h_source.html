35template <typename NumericType>
36template <size_t Num>
37Coefficients<NumericType>& Coefficients<NumericType>::assignImpl (const NumericType* values)
38{
39 static_assert (Num % 2 == 0, "Must supply an even number of coefficients");
40 const auto a0Index = Num / 2;
41 const auto a0 = values[a0Index];
42 const auto a0Inv = a0 != NumericType() ? static_cast<NumericType> (1) / values[a0Index]
43 : NumericType();
44
45 coefficients.clearQuick();
46 coefficients.ensureStorageAllocated ((int) jmax ((size_t) 8, Num));
47
48 for (size_t i = 0; i < Num; ++i)
49 if (i != a0Index)
50 coefficients.add (values[i] * a0Inv);
51
52 return *this;
53}

69template <typename SampleType>
70void Filter<SampleType>::reset (SampleType resetToValue)
71{
72 auto newOrder = coefficients->getFilterOrder();
73
74 if (newOrder != order)
75 {
76 memory.malloc (jmax (order, newOrder, static_cast<size_t> (3)) + 1);
77 state = snapPointerToAlignment (memory.getData(), sizeof (SampleType));
78 order = newOrder;
79 }
80
81 for (size_t i = 0; i < order; ++i)
82 state[i] = resetToValue;
83}

88template <typename SampleType>
89template <typename ProcessContext, bool bypassed>
90void Filter<SampleType>::processInternal (const ProcessContext& context) noexcept
91{
92 static_assert (std::is_same_v<typename ProcessContext::SampleType, SampleType>,
93 "The sample-type of the IIR filter must match the sample-type supplied to this process callback");
94 check();
95
96 auto&& inputBlock = context.getInputBlock();
97 auto&& outputBlock = context.getOutputBlock();
98
99 // This class can only process mono signals. Use the ProcessorDuplicator class
100 // to apply this filter on a multi-channel audio stream.
101 jassert (inputBlock.getNumChannels() == 1);
102 jassert (outputBlock.getNumChannels() == 1);
103
104 auto numSamples = inputBlock.getNumSamples();
105 auto* src = inputBlock .getChannelPointer (0);
106 auto* dst = outputBlock.getChannelPointer (0);
107 auto* coeffs = coefficients->getRawCoefficients();
108
109 switch (order)
110 {
111 case 1:
112 {
113 auto b0 = coeffs[0];
114 auto b1 = coeffs[1];
115 auto a1 = coeffs[2];
116
117 auto lv1 = state[0];
118
119 for (size_t i = 0; i < numSamples; ++i)
120 {
121 auto input = src[i];
122 auto output = input * b0 + lv1;
123
124 dst[i] = bypassed ? input : output;
125
126 lv1 = (input * b1) - (output * a1);
127 }
128
129 util::snapToZero (lv1); state[0] = lv1;
130 }
131 break;
132
133 case 2:
134 {
135 auto b0 = coeffs[0];
136 auto b1 = coeffs[1];
137 auto b2 = coeffs[2];
138 auto a1 = coeffs[3];
139 auto a2 = coeffs[4];
140
141 auto lv1 = state[0];
142 auto lv2 = state[1];
143
144 for (size_t i = 0; i < numSamples; ++i)
145 {
146 auto input = src[i];
147 auto output = (input * b0) + lv1;
148 dst[i] = bypassed ? input : output;
149
150 lv1 = (input * b1) - (output* a1) + lv2;
151 lv2 = (input * b2) - (output* a2);
152 }
153
154 util::snapToZero (lv1); state[0] = lv1;
155 util::snapToZero (lv2); state[1] = lv2;
156 }
157 break;
158
159 case 3:
160 {
161 auto b0 = coeffs[0];
162 auto b1 = coeffs[1];
163 auto b2 = coeffs[2];
164 auto b3 = coeffs[3];
165 auto a1 = coeffs[4];
166 auto a2 = coeffs[5];
167 auto a3 = coeffs[6];
168
169 auto lv1 = state[0];
170 auto lv2 = state[1];
171 auto lv3 = state[2];
172
173 for (size_t i = 0; i < numSamples; ++i)
174 {
175 auto input = src[i];
176 auto output = (input * b0) + lv1;
177 dst[i] = bypassed ? input : output;
178
179 lv1 = (input * b1) - (output* a1) + lv2;
180 lv2 = (input * b2) - (output* a2) + lv3;
181 lv3 = (input * b3) - (output* a3);
182 }
183
184 util::snapToZero (lv1); state[0] = lv1;
185 util::snapToZero (lv2); state[1] = lv2;
186 util::snapToZero (lv3); state[2] = lv3;
187 }
188 break;
189
190 default:
191 {
192 for (size_t i = 0; i < numSamples; ++i)
193 {
194 auto input = src[i];
195 auto output= (input * coeffs[0]) + state[0];
196 dst[i] = bypassed ? input : output;
197
198 for (size_t j = 0; j < order - 1; ++j)
199 state[j] = (input * coeffs[j + 1]) - (output* coeffs[order + j + 1]) + state[j + 1];
200
201 state[order - 1] = (input * coeffs[order]) - (output* coeffs[order * 2]);
202 }
203
204 snapToZero();
205 }
206 }
207}

209template <typename SampleType>
210SampleType JUCE_VECTOR_CALLTYPE Filter<SampleType>::processSample (SampleType sample) noexcept
211{
212 check();
213 auto* c = coefficients->getRawCoefficients();
214
215 auto output = (c[0] * sample) + state[0];
216
217 for (size_t j = 0; j < order - 1; ++j)
218 state[j] = (c[j + 1] * sample) - (c[order + j + 1] * output) + state[j + 1];
219
220 state[order - 1] = (c[order] * sample) - (c[order * 2] * output);
221
222 return output;
223}

225template <typename SampleType>
226void Filter<SampleType>::snapToZero() noexcept
227{
228 for (size_t i = 0; i < order; ++i)
229 util::snapToZero (state[i]);
