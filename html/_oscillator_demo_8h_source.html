57struct OscillatorDemoDSP
58{
59 void prepare (const ProcessSpec& spec)
60 {
61 gain.setGainDecibels (-6.0f);
62
63 for (auto&& oscillator : oscillators)
64 {
65 oscillator.setFrequency (440.f);
66 oscillator.prepare (spec);
67 }
68
69 updateParameters();
70
71 tempBuffer = AudioBlock<float> (tempBufferMemory, spec.numChannels, spec.maximumBlockSize);
72 }
73
74 void process (const ProcessContextReplacing<float>& context)
75 {
76 tempBuffer.copyFrom (context.getInputBlock());
77 tempBuffer.multiplyBy (static_cast<float> (fileMix));
78
79 oscillators[currentOscillatorIdx].process (context);
80 context.getOutputBlock().multiplyBy (static_cast<float> (1.0 - fileMix));
81
82 context.getOutputBlock().add (tempBuffer);
83
84 gain.process (context);
85 }
86
87 void reset()
88 {
89 oscillators[currentOscillatorIdx].reset();
90 }
91
92 void updateParameters()
93 {
94 currentOscillatorIdx = jmin (numElementsInArray (oscillators),
95 3 * (accuracy.getCurrentSelectedID() - 1) + (typeParam.getCurrentSelectedID() - 1));
96
97 auto freq = static_cast<float> (freqParam.getCurrentValue());
98
99 for (auto&& oscillator : oscillators)
100 oscillator.setFrequency (freq);
101
102 gain.setGainDecibels (static_cast<float> (gainParam.getCurrentValue()));
103
104 fileMix = mixParam.getCurrentValue();
105 }
106
107 //==============================================================================
108 Oscillator<float> oscillators[6] =
109 {
110 // No Approximation
111 {[] (float x) { return std::sin (x); }}, // sine
112 {[] (float x) { return x / MathConstants<float>::pi; }}, // saw
113 {[] (float x) { return x < 0.0f ? -1.0f : 1.0f; }}, // square
114
115 // Approximated by a wave-table
116 {[] (float x) { return std::sin (x); }, 100}, // sine
117 {[] (float x) { return x / MathConstants<float>::pi; }, 100}, // saw
118 {[] (float x) { return x < 0.0f ? -1.0f : 1.0f; }, 100} // square
119 };
120
121 int currentOscillatorIdx = 0;
122 Gain<float> gain;
123
124 ChoiceParameter typeParam { {"sine", "saw", "square"}, 1, "Type" };
125 ChoiceParameter accuracy { {"No Approximation", "Use Wavetable"}, 1, "Accuracy" };
126 SliderParameter freqParam { { 20.0, 24000.0 }, 0.4, 440.0, "Frequency", "Hz" };
127 SliderParameter gainParam { { -100.0, 20.0 }, 3.0, -20.0, "Gain", "dB" };
128 SliderParameter mixParam { { 0.0, 1.0 }, 1.0, 0.0, "File mix" };
129
130 HeapBlock<char> tempBufferMemory;
131 AudioBlock<float> tempBuffer;
132 double fileMix;
133
134 std::vector<DSPDemoParameterBase*> parameters { &typeParam, &accuracy, &freqParam, &gainParam, &mixParam };
135};
