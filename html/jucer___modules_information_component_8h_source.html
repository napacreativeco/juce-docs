112 void paintListBoxItem (int rowNumber, Graphics& g, int width, int height, bool rowIsSelected) override
113 {
114 ignoreUnused (height);
115
116 Rectangle<int> bounds (0, 0, width, height);
117
118 g.setColour (rowIsSelected ? findColour (defaultHighlightColourId) : findColour (rowNumber % 2 == 0 ? widgetBackgroundColourId
119 : secondaryWidgetBackgroundColourId));
120 g.fillRect (bounds.withTrimmedBottom (1));
121
122 bounds.removeFromLeft (5);
123 g.setColour (rowIsSelected ? findColour (defaultHighlightedTextColourId) : findColour (widgetTextColourId));
124
125 //==============================================================================
126 auto moduleID = project.getEnabledModules().getModuleID (rowNumber);
127
128 g.drawFittedText (moduleID, bounds.removeFromLeft (roundToInt (listHeader->getProportionAtIndex (0) * (float) width)), Justification::centredLeft, 1);
129
130 //==============================================================================
131 auto version = project.getEnabledModules().getModuleInfo (moduleID).getVersion();
132 if (version.isEmpty())
133 version = "?";
134
135 g.drawFittedText (version, bounds.removeFromLeft (roundToInt (listHeader->getProportionAtIndex (1) * (float) width)), Justification::centredLeft, 1);
136
137 //==============================================================================
138 g.drawFittedText (String (project.getEnabledModules().shouldCopyModuleFilesLocally (moduleID) ? "Yes" : "No"),
139 bounds.removeFromLeft (roundToInt (listHeader->getProportionAtIndex (2) * (float) width)), Justification::centredLeft, 1);
140
141 //==============================================================================
142 String pathText;
143
144 if (project.getEnabledModules().shouldUseGlobalPath (moduleID))
145 {
146 pathText = "Global";
147 }
148 else
149 {
150 StringArray paths;
151
152 for (Project::ExporterIterator exporter (project); exporter.next();)
153 paths.addIfNotAlreadyThere (exporter->getPathForModuleString (moduleID).trim());
154
155 paths.removeEmptyStrings();
156 paths.removeDuplicates (true);
157
158 pathText = paths.joinIntoString (", ");
159 }
160
161 g.drawFittedText (pathText, bounds.removeFromLeft (roundToInt (listHeader->getProportionAtIndex (3) * (float) width)), Justification::centredLeft, 1);
162 }

220 static void setLocalCopyModeForAllModules (Project& project, bool copyLocally)

222 auto& modules = project.getEnabledModules();

278 void showSetPathsMenu()
279 {
280 PopupMenu m;
281 auto moduleToCopy = project.getEnabledModules().getModuleID (list.getSelectedRow());
282
283 if (moduleToCopy.isNotEmpty())
284 {
285 m.addItem (PopupMenu::Item ("Copy the paths from the module '" + moduleToCopy + "' to all other modules")
286 .setAction ([this, moduleToCopy]
287 {
288 auto& modulesList = project.getEnabledModules();
289
290 for (Project::ExporterIterator exporter (project); exporter.next();)
291 {
292 for (int i = 0; i < modulesList.getNumModules(); ++i)
293 {
294 auto modID = modulesList.getModuleID (i);
295
296 if (modID != moduleToCopy)
297 exporter->getPathForModuleValue (modID) = exporter->getPathForModuleValue (moduleToCopy).get();
298 }
299 }
300
301 list.repaint();
302 }));
303
304 m.addItem (PopupMenu::Item ("Copy paths from selected module")
305 .setEnabled (list.getNumSelectedRows() == 1)
306 .setAction ([this, moduleToCopy]
307 {
308 modulePathClipboard.clear();
309
310 for (Project::ExporterIterator exporter (project); exporter.next();)
311 modulePathClipboard[exporter->getUniqueName()] = exporter->getPathForModuleValue (moduleToCopy).get();
312
313 list.repaint();
314 }));
315
316 m.addItem (PopupMenu::Item ("Paste paths to selected modules")
317 .setEnabled (! modulePathClipboard.empty())
318 .setAction ([this]
319 {
320 for (int selectionId = 0; selectionId < list.getNumSelectedRows(); ++selectionId)
321 {
322 auto rowNumber = list.getSelectedRow (selectionId);
323 auto modID = project.getEnabledModules().getModuleID (rowNumber);
324
325 for (Project::ExporterIterator exporter (project); exporter.next();)
326 exporter->getPathForModuleValue (modID) = modulePathClipboard[exporter->getUniqueName()];
327 }
328
329 list.repaint();
330 }));
331 }
332 else
333 {
334 m.addItem (PopupMenu::Item ("(Select a module in the list above to use this option)")
335 .setEnabled (false));
336 }
337
338 m.showMenuAsync (PopupMenu::Options()
339 .withDeletionCheck (*this)
340 .withTargetComponent (copyPathButton));
341 }
