141 template <typename It, typename = ValidIfIterator<It>>
142 void add (It begin, It end)
143 {
144 parameters.reserve (parameters.size() + std::size_t (std::distance (begin, end)));
145 std::transform (std::make_move_iterator (begin),
146 std::make_move_iterator (end),
147 std::back_inserter (parameters),
148 [] (auto item) { return makeParameterStorage (std::move (item)); });
149 }
