43 String uuidPropName ("UUID" + String (windowIndex));
44 clientName = properties.getValue (uuidPropName);

46 if (clientName.isEmpty())

53 setOpaque (true);
54 setSize (1500, 900);
55
56 if (! OSCSender::connect (getBroadcastIPAddress(), clientPortNumber))
57 error = "Client app OSC sender: network connection error.";

59 if (! OSCReceiver::connect (masterPortNumber))

62 OSCReceiver::addListener (this);

64 timerCallback();

74 void mouseDrag (const MouseEvent& e) override
75 {
76 auto clientArea = getAreaInGlobalSpace();
77
78 if (! clientArea.isEmpty())
79 {
80 OSCMessage message (userInputOSCAddress);
81
82 message.addString (clientName);
83 message.addFloat32 (e.position.x * clientArea.getWidth() / (float) getWidth() + clientArea.getX());
84 message.addFloat32 (e.position.y * clientArea.getHeight() / (float) getHeight() + clientArea.getY());

104
105 void canvasStateOSCMessageReceived (const OSCMessage& message)
106 {
107 if (message.isEmpty() || ! message[0].isBlob())
108 return;
109
110 if (packetiser.appendIncomingBlock (message[0].getBlob()))
111 {
112 const ScopedLock sl (canvasLock);
113
114 MemoryBlock newCanvasData;
115
116 if (packetiser.reassemble (newCanvasData))
117 {
118 MemoryInputStream i (newCanvasData.getData(), newCanvasData.getSize(), false);
119 canvas2.load (i);
120 triggerAsyncUpdate();
121 }
122 }
123 }

149 void paint (Graphics& g) override
150 {
151 g.fillAll (canvas.backgroundColour);
152
153 auto clientArea = getAreaInGlobalSpace();
154
155 if (clientArea.isEmpty())
156 {
157 g.setColour (Colours::red.withAlpha (0.5f));
158 g.setFont (20.0f);
159 g.drawText ("Not Connected", getLocalBounds(), Justification::centred, false);
160 return;
161 }
162
163 canvas.draw (g, getLocalBounds().toFloat(), clientArea);
164
165 g.setFont (Font (34.0f));
166 g.setColour (Colours::white.withAlpha (0.6f));
167
168 g.drawText (getMachineInfoToDisplay(),
169 getLocalBounds().reduced (10).removeFromBottom (20),
170 Justification::centredRight, true);
171
172 if (error.isNotEmpty())
173 {
174 g.setColour (Colours::red);
175 g.drawText (error, getLocalBounds().reduced (10).removeFromBottom (80),
176 Justification::centredRight, true);
177 }

180 Rectangle<float> getAreaInGlobalSpace() const
181 {
182 if (auto client = canvas.findClient (clientName))
183 {
184 auto screenBounds = getScreenBounds();
185 auto* display = Desktop::getInstance().getDisplays().getDisplayForPoint (screenBounds.getCentre());
186 return ((screenBounds - display->userArea.getCentre()).toFloat() / (client->scaleFactor * display->dpi / display->scale)) + client->centre;
187 }
188
189 return {};
190 }
191
192 Rectangle<float> getScreenAreaInGlobalSpace() const
193 {
194 if (auto client = canvas.findClient (clientName))
195 {
196 auto* display = Desktop::getInstance().getDisplays().getDisplayForPoint (getScreenBounds().getCentre());
197 return (display->userArea.toFloat() / (client->scaleFactor * display->dpi / display->scale)).withCentre (client->centre);
198 }
199
200 return {};
