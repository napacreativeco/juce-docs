26inline bool b2IsValid(float32 x)
27{
28 if (x != x)
29 {
30 // NaN.
31 return false;
32 }
33
34 float32 infinity = std::numeric_limits<float32>::infinity();
35 return -infinity < x && x < infinity;
36}

39inline float32 b2InvSqrt(float32 x)
40{
41 union
42 {
43 float32 x;
44 juce::int32 i;
45 } convert;
46
47 convert.x = x;
48 float32 xhalf = 0.5f * x;
49 convert.i = 0x5f3759df - (convert.i >> 1);
50 x = convert.x;
51 x = x * (1.5f - xhalf * x * x);
52 return x;
53}

189struct b2Mat22
190{
192 b2Mat22() {}
193
195 b2Mat22(const b2Vec2& c1, const b2Vec2& c2)
196 {
197 ex = c1;
198 ey = c2;
199 }
200
202 b2Mat22(float32 a11, float32 a12, float32 a21, float32 a22)
203 {
204 ex.x = a11; ex.y = a21;
205 ey.x = a12; ey.y = a22;
206 }
207
209 void Set(const b2Vec2& c1, const b2Vec2& c2)
210 {
211 ex = c1;
212 ey = c2;
213 }
214
216 void SetIdentity()
217 {
218 ex.x = 1.0f; ey.x = 0.0f;
219 ex.y = 0.0f; ey.y = 1.0f;
220 }
221
223 void SetZero()
224 {
225 ex.x = 0.0f; ey.x = 0.0f;
226 ex.y = 0.0f; ey.y = 0.0f;
227 }
228
229 b2Mat22 GetInverse() const
230 {
231 float32 a = ex.x, b = ey.x, c = ex.y, d = ey.y;
232 b2Mat22 B;
233 float32 det = a * d - b * c;
234 if (det != 0.0f)
235 {
236 det = 1.0f / det;
237 }
238 B.ex.x = det * d; B.ey.x = -det * b;
239 B.ex.y = -det * c; B.ey.y = det * a;
240 return B;
241 }
242
245 b2Vec2 Solve(const b2Vec2& b) const
246 {
247 float32 a11 = ex.x, a12 = ey.x, a21 = ex.y, a22 = ey.y;
248 float32 det = a11 * a22 - a12 * a21;
249 if (det != 0.0f)
250 {
251 det = 1.0f / det;
252 }
253 b2Vec2 x;
254 x.x = det * (a22 * b.x - a12 * b.y);
255 x.y = det * (a11 * b.y - a21 * b.x);
256 return x;
257 }
258
259 b2Vec2 ex, ey;
260};

546inline b2Rot b2Mul(const b2Rot& q, const b2Rot& r)
547{
548 // [qc -qs] * [rc -rs] = [qc*rc-qs*rs -qc*rs-qs*rc]
549 // [qs qc] [rs rc] [qs*rc+qc*rs -qs*rs+qc*rc]
550 // s = qs * rc + qc * rs
551 // c = qc * rc - qs * rs
552 b2Rot qr;
553 qr.s = q.s * r.c + q.c * r.s;
554 qr.c = q.c * r.c - q.s * r.s;
555 return qr;
556}
557
559inline b2Rot b2MulT(const b2Rot& q, const b2Rot& r)
560{
561 // [ qc qs] * [rc -rs] = [qc*rc+qs*rs -qc*rs+qs*rc]
562 // [-qs qc] [rs rc] [-qs*rc+qc*rs qs*rs+qc*rc]
563 // s = qc * rs - qs * rc
564 // c = qc * rc + qs * rs
565 b2Rot qr;
566 qr.s = q.c * r.s - q.s * r.c;
567 qr.c = q.c * r.c + q.s * r.s;
568 return qr;
569}

583inline b2Vec2 b2Mul(const b2Transform& T, const b2Vec2& v)
584{
585 float32 x = (T.q.c * v.x - T.q.s * v.y) + T.p.x;
586 float32 y = (T.q.s * v.x + T.q.c * v.y) + T.p.y;
587
588 return b2Vec2(x, y);
589}

591inline b2Vec2 b2MulT(const b2Transform& T, const b2Vec2& v)
592{
593 float32 px = v.x - T.p.x;
594 float32 py = v.y - T.p.y;
595 float32 x = (T.q.c * px + T.q.s * py);
596 float32 y = (-T.q.s * px + T.q.c * py);
597
598 return b2Vec2(x, y);
599}

601// v2 = A.q.Rot(B.q.Rot(v1) + B.p) + A.p
602// = (A.q * B.q).Rot(v1) + A.q.Rot(B.p) + A.p
603inline b2Transform b2Mul(const b2Transform& A, const b2Transform& B)
604{
605 b2Transform C;
606 C.q = b2Mul(A.q, B.q);
607 C.p = b2Mul(A.q, B.p) + A.p;
608 return C;
609}
610
611// v2 = A.q' * (B.q * v1 + B.p - A.p)
612// = A.q' * B.q * v1 + A.q' * (B.p - A.p)
613inline b2Transform b2MulT(const b2Transform& A, const b2Transform& B)
614{
615 b2Transform C;
616 C.q = b2MulT(A.q, B.q);
617 C.p = b2MulT(A.q, B.p - A.p);
618 return C;
619}

698inline void b2Sweep::GetTransform(b2Transform* xf, float32 beta) const
699{
700 xf->p = (1.0f - beta) * c0 + beta * c;
701 float32 angle = (1.0f - beta) * a0 + beta * a;
702 xf->q.Set(angle);
703
704 // Shift to origin
705 xf->p -= b2Mul(xf->q, localCenter);
706}

708inline void b2Sweep::Advance(float32 alpha)
709{
710 b2Assert(alpha0 < 1.0f);
711 float32 beta = (alpha - alpha0) / (1.0f - alpha0);
712 c0 = (1.0f - beta) * c0 + beta * c;
713 a0 = (1.0f - beta) * a0 + beta * a;
714 alpha0 = alpha;
715}

718inline void b2Sweep::Normalize()
719{
720 float32 twoPi = 2.0f * b2_pi;
721 float32 d = twoPi * floorf(a0 / twoPi);
722 a0 -= d;
723 a -= d;
724}
