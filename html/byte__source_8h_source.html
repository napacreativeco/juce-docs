73static inline SERD_PURE_FUNC uint8_t
74serd_byte_source_peek(SerdByteSource* source)
75{
76 assert(source->prepared);
77 return source->read_buf[source->read_head];
78}

80static inline SerdStatus
81serd_byte_source_advance(SerdByteSource* source)
82{
83 SerdStatus st = SERD_SUCCESS;
84
85 switch (serd_byte_source_peek(source)) {
86 case '\n':
87 ++source->cur.line;
88 source->cur.col = 0;
89 break;
90 default:
91 ++source->cur.col;
92 }
93
94 const bool was_eof = source->eof;
95 if (source->from_stream) {
96 source->eof = false;
97 if (source->page_size > 1) {
98 if (++source->read_head == source->page_size) {
99 st = serd_byte_source_page(source);
100 } else if (source->read_head == source->buf_size) {
101 source->eof = true;
102 }
103 } else {
104 if (!source->read_func(&source->read_byte, 1, 1, source->stream)) {
105 source->eof = true;
106 st =
107 source->error_func(source->stream) ? SERD_ERR_UNKNOWN : SERD_FAILURE;
108 }
109 }
110 } else if (!source->eof) {
111 ++source->read_head; // Move to next character in string
112 if (source->read_buf[source->read_head] == '\0') {
113 source->eof = true;
114 }
115 }
116
117 return (was_eof && source->eof) ? SERD_FAILURE : st;
118}
