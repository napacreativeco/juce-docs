63template <typename... Processors>
64class ProcessorChain
65{
66public:
68 template <int Index> auto& get() noexcept { return std::get<Index> (processors); }
69
71 template <int Index> const auto& get() const noexcept { return std::get<Index> (processors); }
72
74 template <int Index>
75 void setBypassed (bool b) noexcept { bypassed[(size_t) Index] = b; }
76
78 template <int Index>
79 bool isBypassed() const noexcept { return bypassed[(size_t) Index]; }
80
82 void prepare (const ProcessSpec& spec)
83 {
84 detail::forEachInTuple ([&] (auto& proc, auto) { proc.prepare (spec); }, processors);
85 }
86
88 void reset()
89 {
90 detail::forEachInTuple ([] (auto& proc, auto) { proc.reset(); }, processors);
91 }
92
94 template <typename ProcessContext>
95 void process (const ProcessContext& context) noexcept
96 {
97 detail::forEachInTuple ([this, &context] (auto& proc, auto index) noexcept { this->processOne (context, proc, index); },
98 processors);
99 }
100
101private:
102 template <typename Context, typename Proc, size_t Ix>
103 void processOne (const Context& context, Proc& proc, std::integral_constant<size_t, Ix>) noexcept
104 {
105 if constexpr (detail::useContextDirectly<Context, Ix>)
106 {
107 auto contextCopy = context;
108 contextCopy.isBypassed = (bypassed[Ix] || context.isBypassed);
109
110 proc.process (contextCopy);
111 }
112 else
113 {
114 jassert (context.getOutputBlock().getNumChannels() == context.getInputBlock().getNumChannels());
115 ProcessContextReplacing<typename Context::SampleType> replacingContext (context.getOutputBlock());
116 replacingContext.isBypassed = (bypassed[Ix] || context.isBypassed);
117
118 proc.process (replacingContext);
119 }
120 }
121
122 std::tuple<Processors...> processors;
123 std::array<bool, sizeof...(Processors)> bypassed { {} };
124};
