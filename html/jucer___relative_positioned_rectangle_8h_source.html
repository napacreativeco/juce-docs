281 void setModes (const AnchorPoint xAnchor, const PositionMode xMode_,
282 const AnchorPoint yAnchor, const PositionMode yMode_,
283 const SizeMode widthMode, const SizeMode heightMode,
284 const Rectangle<int>& target) noexcept
285 {
286 if (xMode != ((int) xAnchor | (int) xMode_) || wMode != widthMode)

290

292 wMode = (uint8) widthMode;

295 }
296

303 hMode = (uint8) heightMode;
304
305 updatePosAndSize (y, h, ty, th, yMode, hMode, target.getY(), target.getHeight());

440 void addPosDescription (String& s, const uint8 mode, const double value) const noexcept
441 {
442 if ((mode & proportionOfParentSize) != 0)
443 {
444 s << (roundToInt (value * 100000.0) / 1000.0) << '%';
445 }
446 else
447 {
448 s << (roundToInt (value * 100.0) / 100.0);
449
450 if ((mode & absoluteFromParentBottomRight) != 0)
451 s << 'R';
452 else if ((mode & absoluteFromParentCentre) != 0)
453 s << 'C';
454 }
455
456 if ((mode & anchorAtRightOrBottom) != 0)
457 s << 'r';
458 else if ((mode & anchorAtCentre) != 0)
459 s << 'c';
460 }
461
462 void addSizeDescription (String& s, const uint8 mode, const double value) const noexcept
463 {
464 if (mode == proportionalSize)
465 s << (roundToInt (value * 100000.0) / 1000.0) << '%';
466 else if (mode == parentSizeMinusAbsolute)
467 s << (roundToInt (value * 100.0) / 100.0) << 'M';
468 else
469 s << (roundToInt (value * 100.0) / 100.0);
470 }
471
472 void decodePosString (const String& s, uint8& mode, double& value) noexcept
473 {
474 if (s.containsChar ('r'))
475 mode = anchorAtRightOrBottom;
476 else if (s.containsChar ('c'))
477 mode = anchorAtCentre;
478 else
479 mode = anchorAtLeftOrTop;
480
481 if (s.containsChar ('%'))
482 {
483 mode |= proportionOfParentSize;
484 value = s.removeCharacters ("%rcRC").getDoubleValue() / 100.0;
485 }
486 else
487 {
488 if (s.containsChar ('R'))
489 mode |= absoluteFromParentBottomRight;
490 else if (s.containsChar ('C'))
491 mode |= absoluteFromParentCentre;
492 else
493 mode |= absoluteFromParentTopLeft;
494
495 value = s.removeCharacters ("rcRC").getDoubleValue();
496 }
497 }
498
499 void decodeSizeString (const String& s, uint8& mode, double& value) noexcept
500 {
501 if (s.containsChar ('%'))
502 {
503 mode = proportionalSize;
504 value = s.upToFirstOccurrenceOf ("%", false, false).getDoubleValue() / 100.0;
505 }
506 else if (s.containsChar ('M'))
507 {
508 mode = parentSizeMinusAbsolute;
509 value = s.getDoubleValue();
510 }
511 else
512 {
513 mode = absoluteSize;
514 value = s.getDoubleValue();
515 }
516 }
517
518 void applyPosAndSize (double& xOut, double& wOut, const double x_, const double w_,
519 const uint8 xMode_, const uint8 wMode_,
520 const int parentPos, const int parentSize) const noexcept
521 {
522 if (wMode_ == proportionalSize)
523 wOut = roundToInt (w_ * parentSize);
524 else if (wMode_ == parentSizeMinusAbsolute)
525 wOut = jmax (0, parentSize - roundToInt (w_));
526 else
527 wOut = roundToInt (w_);
528
529 if ((xMode_ & proportionOfParentSize) != 0)
530 xOut = parentPos + x_ * parentSize;
531 else if ((xMode_ & absoluteFromParentBottomRight) != 0)
532 xOut = (parentPos + parentSize) - x_;
533 else if ((xMode_ & absoluteFromParentCentre) != 0)
534 xOut = x_ + (parentPos + parentSize / 2);
535 else
536 xOut = x_ + parentPos;
537
538 if ((xMode_ & anchorAtRightOrBottom) != 0)
539 xOut -= wOut;
540 else if ((xMode_ & anchorAtCentre) != 0)
541 xOut -= wOut / 2;
542 }
543
544 void updatePosAndSize (double& xOut, double& wOut, double x_, const double w_,
545 const uint8 xMode_, const uint8 wMode_,
546 const int parentPos, const int parentSize) const noexcept
547 {
548 if (wMode_ == proportionalSize)
549 {
550 if (parentSize > 0)
551 wOut = w_ / parentSize;
552 }
553 else if (wMode_ == parentSizeMinusAbsolute)
554 wOut = parentSize - w_;
555 else
556 wOut = w_;
557
558 if ((xMode_ & anchorAtRightOrBottom) != 0)
559 x_ += w_;
560 else if ((xMode_ & anchorAtCentre) != 0)
561 x_ += w_ / 2;
562
563 if ((xMode_ & proportionOfParentSize) != 0)
564 {
565 if (parentSize > 0)
566 xOut = (x_ - parentPos) / parentSize;
567 }
568 else if ((xMode_ & absoluteFromParentBottomRight) != 0)
569 xOut = (parentPos + parentSize) - x_;
570 else if ((xMode_ & absoluteFromParentCentre) != 0)
571 xOut = x_ - (parentPos + parentSize / 2);
572 else
573 xOut = x_ - parentPos;
574 }

623 template <typename LayoutType>
624 void getRelativeTargetBounds (const Rectangle<int>& parentArea,
625 const LayoutType* layout,
626 int& x, int& xw, int& y, int& yh,
627 int& w, int& h) const
628 {
629 Component* rx = {};
630 Component* ry = {};
631 Component* rw = {};
632 Component* rh = {};
633
634 if (layout != nullptr)
635 {
636 rx = layout->findComponentWithId (relativeToX);
637 ry = layout->findComponentWithId (relativeToY);
638 rw = layout->findComponentWithId (relativeToW);
639 rh = layout->findComponentWithId (relativeToH);
640 }
641
642 x = parentArea.getX() + (rx != nullptr ? rx->getX() : 0);
643 y = parentArea.getY() + (ry != nullptr ? ry->getY() : 0);
644 w = rw != nullptr ? rw->getWidth() : parentArea.getWidth();
645 h = rh != nullptr ? rh->getHeight() : parentArea.getHeight();
646 xw = rx != nullptr ? rx->getWidth() : parentArea.getWidth();
647 yh = ry != nullptr ? ry->getHeight() : parentArea.getHeight();
648 }
649
650 Rectangle<int> getRectangle (const Rectangle<int>& parentArea,
651 const ComponentLayout* layout) const
652 {
653 int x, xw, y, yh, w, h;
654 getRelativeTargetBounds (parentArea, layout, x, xw, y, yh, w, h);
655
656 const Rectangle<int> xyRect ((xw <= 0 || yh <= 0) ? Rectangle<int>()
657 : rect.getRectangle (Rectangle<int> (x, y, xw, yh)));
658
659 const Rectangle<int> whRect ((w <= 0 || h <= 0) ? Rectangle<int>()
660 : rect.getRectangle (Rectangle<int> (x, y, w, h)));
661
662 return Rectangle<int> (xyRect.getX(), xyRect.getY(),
663 whRect.getWidth(), whRect.getHeight());
664 }

676 }
