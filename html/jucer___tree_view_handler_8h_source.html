30class TreeViewHandler : public ComponentTypeHandler
31{
32public:
33 TreeViewHandler()
34 : ComponentTypeHandler ("TreeView", "juce::TreeView", typeid (DemoTreeView), 150, 150)
35 {
36 registerColour (juce::TreeView::backgroundColourId, "background", "backgroundColour");
37 registerColour (juce::TreeView::linesColourId, "lines", "linecol");
38 }
39
40 Component* createNewComponent (JucerDocument*) override
41 {
42 return new DemoTreeView();
43 }
44
45 XmlElement* createXmlFor (Component* comp, const ComponentLayout* layout) override
46 {
47 TreeView* const t = dynamic_cast<TreeView*> (comp);
48 XmlElement* const e = ComponentTypeHandler::createXmlFor (comp, layout);
49
50 e->setAttribute ("rootVisible", t->isRootItemVisible());
51 e->setAttribute ("openByDefault", t->areItemsOpenByDefault());
52
53 return e;
54 }
55
56 bool restoreFromXml (const XmlElement& xml, Component* comp, const ComponentLayout* layout) override
57 {
58 if (! ComponentTypeHandler::restoreFromXml (xml, comp, layout))
59 return false;
60
61 TreeView defaultTreeView;
62 TreeView* const t = dynamic_cast<TreeView*> (comp);
63
64 t->setRootItemVisible (xml.getBoolAttribute ("rootVisible", defaultTreeView.isRootItemVisible()));
65 t->setDefaultOpenness (xml.getBoolAttribute ("openByDefault", defaultTreeView.areItemsOpenByDefault()));
66
67 return true;
68 }
69
70 void getEditableProperties (Component* component, JucerDocument& document,
71 Array<PropertyComponent*>& props, bool multipleSelected) override
72 {
73 ComponentTypeHandler::getEditableProperties (component, document, props, multipleSelected);
74
75 if (multipleSelected)
76 return;
77
78 auto* t = dynamic_cast<TreeView*> (component);
79
80 props.add (new TreeViewRootItemProperty (t, document));
81 props.add (new TreeViewRootOpennessProperty (t, document));
82
83 addColourProperties (t, document, props);
84 }
85
86 String getCreationParameters (GeneratedCode&, Component* comp) override
87 {
88 return quotedString (comp->getName(), false);
89 }
90
91 void fillInCreationCode (GeneratedCode& code, Component* component, const String& memberVariableName) override
92 {
93 TreeView defaultTreeView;
94 TreeView* const t = dynamic_cast<TreeView*> (component);
95
96 ComponentTypeHandler::fillInCreationCode (code, component, memberVariableName);
97
98 if (defaultTreeView.isRootItemVisible() != t->isRootItemVisible())
99 {
100 code.constructorCode
101 << memberVariableName << "->setRootItemVisible ("
102 << CodeHelpers::boolLiteral (t->isRootItemVisible()) << ");\n";
103 }
104
105 if (defaultTreeView.areItemsOpenByDefault() != t->areItemsOpenByDefault())
106 {
107 code.constructorCode
108 << memberVariableName << "->setDefaultOpenness ("
109 << CodeHelpers::boolLiteral (t->areItemsOpenByDefault()) << ");\n";
110 }
111
112 code.constructorCode << getColourIntialisationCode (component, memberVariableName);
113 code.constructorCode << "\n";
114 }

131
132 private:
133 class DemoTreeViewItem : public TreeViewItem
134 {
135 public:
136 DemoTreeViewItem (const String& name_, const int numItems)
137 : name (name_)
138 {
139 for (int i = 0; i < numItems; ++i)
140 addSubItem (new DemoTreeViewItem ("Demo sub-node " + String (i), numItems - 1));
141 }
142
143 void paintItem (Graphics& g, int width, int height) override
144 {
145 if (isSelected())
146 g.fillAll (Colours::lightblue);
147
148 g.setColour (Colours::black);
149 g.setFont ((float) height * 0.7f);
150 g.drawText (name, 4, 0, width - 4, height, Justification::centredLeft, true);
151 }
152
153 bool mightContainSubItems() override
154 {
155 return true;
156 }
157
158 const String name;
159 };
