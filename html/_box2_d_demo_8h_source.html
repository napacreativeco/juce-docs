116 void paintListBoxItem (int row, Graphics& g, int w, int h, bool rowIsSelected) override
117 {
118 auto& lf = LookAndFeel::getDefaultLookAndFeel();
119
120 if (rowIsSelected)
121 g.fillAll (Colour::contrasting (lf.findColour (ListBox::textColourId),
122 lf.findColour (ListBox::backgroundColourId)));
123
124 g.setColour (lf.findColour (ListBox::textColourId));
125 g.setFont ((float) h * 0.7f);
126 g.drawText (tests[row], Rectangle<int> (0, 0, w, h).reduced (2),
127 Justification::centredLeft, true);
128 }

163 private Timer,
164 private ChangeListener

176 Box2DDemo()
177 : testsList (getTestsList())
178 {
179 setOpaque (true);
180 setWantsKeyboardFocus (true);
181
182 testsListModel.addChangeListener (this);
183
184 addAndMakeVisible (renderComponent);
185
186 addAndMakeVisible (testsListBox);
187 testsListBox.setModel (&testsListModel);
188 testsListBox.selectRow (dominoes);
189
190 addAndMakeVisible (instructions);
191 instructions.setMultiLine (true);
192 instructions.setReadOnly (true);
193
194 startTimerHz (60);
195
196 setSize (500, 500);
197 }
198
199 ~Box2DDemo() override
200 {
201 testsListModel.removeChangeListener (this);
202 }
203
204 void paint (Graphics& g) override
205 {
206 g.fillAll (getUIColourIfAvailable (LookAndFeel_V4::ColourScheme::UIColour::windowBackground));
207 }
208
209 void resized() override
210 {
211 auto r = getLocalBounds().reduced (4);
212
213 auto area = r.removeFromBottom (150);
214 testsListBox.setBounds (area.removeFromLeft (150));
215
216 area.removeFromLeft (4);
217 instructions.setBounds (area);
218
219 r.removeFromBottom (6);
220 renderComponent.setBounds (r);
221 }
222
223 bool keyPressed (const KeyPress& key) override
224 {
225 if (renderComponent.currentTest.get() != nullptr)
226 {
227 // We override this to avoid the system beeping for an unused keypress
228 switch (key.getTextCharacter())
229 {
230 case 'a':
231 case 'w':
232 case 'd':
233 return true;
234
235 default:
236 break;

238 }

280 void checkKeys()

295
296 void timerCallback() override
297 {
298 if (renderComponent.currentTest.get() == nullptr)
299 return;
300
301 if (isShowing())
302 grabKeyboardFocus();
303
304 checkKeys();
305 renderComponent.currentTest->m_world->Step (1.0f / 60.0f, 6, 2);
306 repaint();
307 }
308
309 void changeListenerCallback (ChangeBroadcaster* source) override
310 {

315 renderComponent.currentTest.reset (createTest (index));
316 instructions.setText (getInstructions (index));
317
318 repaint();
319 }
320 }
