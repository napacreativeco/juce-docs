52enum KeyPressCommandIDs
53{
54 buttonMoveUp = 1,
55 buttonMoveRight,
56 buttonMoveDown,
57 buttonMoveLeft,
58 nextButtonColour,
59 previousButtonColour,
60 nextBackgroundColour,
61 previousBackgroundColour
62};

69class KeyPressTarget : public Component,
70 public ApplicationCommandTarget
71{
72public:
73 KeyPressTarget()
74 {
75 Array<Colour> coloursToUse { Colours::darkblue, Colours::darkgrey, Colours::red,
76 Colours::green, Colours::blue, Colours::hotpink };
77 colours.addArray (coloursToUse);
78
79 addAndMakeVisible (button);
80 }
81
82 //==============================================================================
83 void resized() override
84 {
85 auto bounds = getLocalBounds();
86
87 // keep the button on-screen
88 if (buttonX < -150 || buttonX > bounds.getWidth()
89 || buttonY < -30 || buttonY > bounds.getHeight())
90 {
91 buttonX = bounds.getCentreX() - 75;
92 buttonY = bounds.getCentreY() - 15;
93 }
94
95 button.setBounds (buttonX, buttonY, 150, 30);
96 }
97
98 void paint (Graphics& g) override
99 {
100 g.fillAll (colours.getUnchecked (backgroundColourIndex));
101 }
102
103 //==============================================================================
105 ApplicationCommandTarget* getNextCommandTarget() override { return nullptr; }
106
107 void getAllCommands (Array<CommandID>& commands) override
108 {
109 Array<CommandID> ids { KeyPressCommandIDs::buttonMoveUp, KeyPressCommandIDs::buttonMoveRight,
110 KeyPressCommandIDs::buttonMoveDown, KeyPressCommandIDs::buttonMoveLeft,
111 KeyPressCommandIDs::nextButtonColour, KeyPressCommandIDs::previousButtonColour,
112 KeyPressCommandIDs::nextBackgroundColour, KeyPressCommandIDs::previousBackgroundColour };
113
114 commands.addArray (ids);
115 }
116
117 void getCommandInfo (CommandID commandID, ApplicationCommandInfo& result) override
118 {
119 switch (commandID)
120 {
121 case KeyPressCommandIDs::buttonMoveUp:
122 result.setInfo ("Move up", "Move the button up", "Button", 0);
123 result.addDefaultKeypress (KeyPress::upKey, 0);
124 break;
125 case KeyPressCommandIDs::buttonMoveRight:
126 result.setInfo ("Move right", "Move the button right", "Button", 0);
127 result.addDefaultKeypress (KeyPress::rightKey, 0);
128 break;
129 case KeyPressCommandIDs::buttonMoveDown:
130 result.setInfo ("Move down", "Move the button down", "Button", 0);
131 result.addDefaultKeypress (KeyPress::downKey, 0);
132 break;
133 case KeyPressCommandIDs::buttonMoveLeft:
134 result.setInfo ("Move left", "Move the button left", "Button", 0);
135 result.addDefaultKeypress (KeyPress::leftKey, 0);
136 break;
137 case KeyPressCommandIDs::nextButtonColour:
138 result.setInfo ("Next colour", "Change the colour of the button to the next in the list", "Button", 0);
139 result.addDefaultKeypress (KeyPress::rightKey, ModifierKeys::shiftModifier);
140 break;
141 case KeyPressCommandIDs::previousButtonColour:
142 result.setInfo ("Previous colour", "Change the colour of the button to the previous in the list", "Button", 0);
143 result.addDefaultKeypress (KeyPress::leftKey, ModifierKeys::shiftModifier);
144 break;
145 case KeyPressCommandIDs::nextBackgroundColour:
146 result.setInfo ("Next colour", "Change the colour of the background to the next in the list", "Other", 0);
147 result.addDefaultKeypress (KeyPress::rightKey, ModifierKeys::commandModifier);
148 break;
149 case KeyPressCommandIDs::previousBackgroundColour:
150 result.setInfo ("Previous colour", "Change the colour of the background to the previous in the list", "Other", 0);
151 result.addDefaultKeypress (KeyPress::leftKey, ModifierKeys::commandModifier);
152 break;
153 default:
154 break;
155 }
156 }
157
158 bool perform (const InvocationInfo& info) override
159 {
160 switch (info.commandID)
161 {
162 case KeyPressCommandIDs::buttonMoveUp:
163 buttonY -= 5;
164 resized();
165 break;
166 case KeyPressCommandIDs::buttonMoveRight:
167 buttonX += 5;
168 resized();
169 break;
170 case KeyPressCommandIDs::buttonMoveDown:
171 buttonY += 5;
172 resized();
173 break;
174 case KeyPressCommandIDs::buttonMoveLeft:
175 buttonX -= 5;
176 resized();
177 break;
178 case KeyPressCommandIDs::nextButtonColour:
179 ++buttonColourIndex %= colours.size();
180 button.setColour (TextButton::buttonColourId, colours.getUnchecked (buttonColourIndex));
181 break;
182 case KeyPressCommandIDs::previousButtonColour:
183 --buttonColourIndex;
184 if (buttonColourIndex < 0)
185 buttonColourIndex = colours.size() - 1;
186 button.setColour (TextButton::buttonColourId, colours.getUnchecked (buttonColourIndex));
187 break;
188 case KeyPressCommandIDs::nextBackgroundColour:
189 ++backgroundColourIndex %= colours.size();
190 repaint();
191 break;
192 case KeyPressCommandIDs::previousBackgroundColour:
193 --backgroundColourIndex;
194 if (backgroundColourIndex < 0)
195 backgroundColourIndex = colours.size() - 1;
196 repaint();
197 break;
198 default:
199 return false;
200 }
201
202 return true;
203 }
204
205private:
206 TextButton button;
207 int buttonX = -200, buttonY = -200;
208
209 Array<Colour> colours;
210
211 int buttonColourIndex = 0;
212 int backgroundColourIndex = 1;
213};

216class KeyMappingsDemo : public Component
217{
218public:
219 KeyMappingsDemo()
220 {
221 // register the commands that the target component can perform
222 commandManager.registerAllCommandsForTarget (&keyTarget);
223
224 setOpaque (true);
225 addAndMakeVisible (keyMappingEditor);
226 addAndMakeVisible (keyTarget);
227
228 // add command manager key mappings as a KeyListener to the top-level component
229 // so it is notified of key presses
230 getTopLevelComponent()->addKeyListener (commandManager.getKeyMappings());
231
232 setSize (500, 500);
233
234 Timer::callAfterDelay (300, [this] { keyTarget.grabKeyboardFocus(); }); // ensure that key presses are sent to the KeyPressTarget object
235 }
236
237 void paint (Graphics& g) override
238 {
239 g.fillAll (getUIColourIfAvailable (LookAndFeel_V4::ColourScheme::UIColour::windowBackground,
240 Colour::greyLevel (0.93f)));
241 }
242
243 void resized() override
244 {
245 auto bounds = getLocalBounds();
246
247 keyTarget .setBounds (bounds.removeFromTop (bounds.getHeight() / 2).reduced (4));
248 keyMappingEditor.setBounds (bounds.reduced (4));
249 }
250
251private:
252 #if JUCE_DEMO_RUNNER
253 ApplicationCommandManager& commandManager = getGlobalCommandManager();
254 #else
255 ApplicationCommandManager commandManager;
256 #endif
257
258 KeyMappingEditorComponent keyMappingEditor { *commandManager.getKeyMappings(), true};
259
260 KeyPressTarget keyTarget;
261
262 void lookAndFeelChanged() override
263 {
264 auto* lf = &LookAndFeel::getDefaultLookAndFeel();
265
266 keyMappingEditor.setColours (lf->findColour (KeyMappingEditorComponent::backgroundColourId),
267 lf->findColour (KeyMappingEditorComponent::textColourId));
268 }
