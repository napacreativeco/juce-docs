51 static CodeBlocksProjectExporter* createForSettings (Project& projectToUse, const ValueTree& settingsToUse)
52 {
53 // this will also import legacy jucer files where CodeBlocks only worked for Windows,
54 // had valueTreetTypeName "CODEBLOCKS", and there was no OS distinction
55 if (settingsToUse.hasType (getValueTreeTypeNameWindows()) || settingsToUse.hasType ("CODEBLOCKS"))

57
58 if (settingsToUse.hasType (getValueTreeTypeNameLinux()))
59 return new CodeBlocksProjectExporter (projectToUse, settingsToUse, linuxTarget);
60
61 return nullptr;
62 }

325 StringArray getDefines (const BuildConfiguration& config, CodeBlocksTarget& target) const
326 {
327 StringPairArray defines;
328
329 if (isWindows())
330 {
331 defines.set ("__MINGW__", "1");
332 defines.set ("__MINGW_EXTENSION", {});
333
334 auto targetPlatform = getTargetPlatformString();
335
336 if (targetPlatform.isNotEmpty())
337 defines.set ("WINVER", targetPlatform);
338 }
339 else
340 {
341 defines.set ("LINUX", "1");
342 }
343
344 if (config.isDebug())
345 {
346 defines.set ("DEBUG", "1");
347 defines.set ("_DEBUG", "1");
348 }
349 else
350 {
351 defines.set ("NDEBUG", "1");
352 }
353
354 defines = mergePreprocessorDefs (defines, getAllPreprocessorDefs (config, target.type));
355
356 StringArray defs;
357 auto keys = defines.getAllKeys();
358 auto values = defines.getAllValues();
359
360 const auto escapedQuote = isWindows() ? "\\\"" : "\\\\\"";
361
362 for (int i = 0; i < defines.size(); ++i)
363 {
364 auto result = keys[i];
365
366 if (values[i].isNotEmpty())
367 result += "=\"" + values[i].replace ("\"", escapedQuote) + "\"";
368
369 defs.add (result);
370 }
371
372 return getCleanedStringArray (defs);
373 }

375 StringArray getCompilerFlags (const BuildConfiguration& config, CodeBlocksTarget& target) const
376 {
377 StringArray flags;
378
379 if (auto* codeBlocksConfig = dynamic_cast<const CodeBlocksBuildConfiguration*> (&config))
380 flags.add (codeBlocksConfig->getArchitectureTypeString());
381
382 auto recommendedFlags = config.getRecommendedCompilerWarningFlags();
383
384 for (auto& recommendedFlagsType : { recommendedFlags.common, recommendedFlags.cpp })
385 for (auto& recommended : recommendedFlagsType)
386 flags.add (recommended);
387
388 flags.add ("-O" + config.getGCCOptimisationFlag());
389
390 if (config.isLinkTimeOptimisationEnabled())
391 flags.add ("-flto");
392
393 {
394 auto cppStandard = config.project.getCppStandardString();
395
396 if (cppStandard == "latest")
397 cppStandard = project.getLatestNumberedCppStandardString();
398
399 flags.add ("-std=" + String (shouldUseGNUExtensions() ? "gnu++" : "c++") + cppStandard);
400 }
401
402 flags.add ("-mstackrealign");
403
404 if (config.isDebug())
405 flags.add ("-g");
406
407 flags.addTokens (replacePreprocessorTokens (config, config.getAllCompilerFlagsString()).trim(),
408 " \n", "\"'");
409
410 if (config.exporter.isLinux())
411 {
412 if (target.isDynamicLibrary() || getProject().isAudioPluginProject())
413 flags.add ("-fPIC");
414
415 auto packages = config.exporter.getLinuxPackages (PackageDependencyType::compile);
416
417 if (! packages.isEmpty())
418 {
419 auto pkgconfigFlags = String ("`pkg-config --cflags");
420
421 for (auto& p : packages)
422 pkgconfigFlags << " " << p;
423
424 pkgconfigFlags << "`";
425 flags.add (pkgconfigFlags);
426 }
427
428 if (linuxLibs.contains ("pthread"))
429 flags.add ("-pthread");
430 }
431
432 return getCleanedStringArray (flags);
433 }
434
435 StringArray getLinkerFlags (const BuildConfiguration& config, CodeBlocksTarget& target) const
436 {
437 auto flags = makefileExtraLinkerFlags;
438
439 if (auto* codeBlocksConfig = dynamic_cast<const CodeBlocksBuildConfiguration*> (&config))
440 flags.add (codeBlocksConfig->getArchitectureTypeString());
441
442 if (! config.isDebug())
443 flags.add ("-s");
444
445 if (config.isLinkTimeOptimisationEnabled())
446 flags.add ("-flto");
447
448 flags.addTokens (replacePreprocessorTokens (config, config.getAllLinkerFlagsString()).trim(), " \n", "\"'");
449
450 if (config.exporter.isLinux())
451 {
452 if (target.isDynamicLibrary())
453 flags.add ("-shared");
454
455 auto packages = config.exporter.getLinuxPackages (PackageDependencyType::link);
456
457 if (! packages.isEmpty())
458 {
459 String pkgconfigLibs ("`pkg-config --libs");
460
461 for (auto& p : packages)
462 pkgconfigLibs << " " << p;
463
464 pkgconfigLibs << "`";
465 flags.add (pkgconfigLibs);
466 }
467 }
468
469 return getCleanedStringArray (flags);
470 }

482 StringArray getIncludePaths (const BuildConfiguration& config) const
483 {
484 StringArray paths;
485
486 paths.add (".");
487 paths.addArray (extraSearchPaths);
488 paths.addArray (config.getHeaderSearchPaths());
489
490 if (! isWindows())
491 {
492 paths.add ("/usr/include/freetype2");
493
494 // Replace ~ character with $(HOME) environment variable
495 for (auto& path : paths)
496 path = path.replace ("~", "$(HOME)");
497 }
498
499 return getCleanedStringArray (paths);

512 String getOutputPathForTarget (CodeBlocksTarget& target, const BuildConfiguration& config) const
513 {
514 String outputPath;
515 if (config.getTargetBinaryRelativePathString().isNotEmpty())
516 {
517 build_tools::RelativePath binaryPath (config.getTargetBinaryRelativePathString(), build_tools::RelativePath::projectFolder);
518 binaryPath = binaryPath.rebased (projectFolder, getTargetFolder(), build_tools::RelativePath::buildTargetFolder);
519 outputPath = config.getTargetBinaryRelativePathString();
520 }
521 else
522 {
523 outputPath ="bin/" + File::createLegalFileName (config.getName().trim());
524 }
525
526 return outputPath + "/" + replacePreprocessorTokens (config, config.getTargetBinaryNameString() + target.getTargetSuffix());
527 }

529 String getSharedCodePath (const BuildConfiguration& config) const
530 {

532 build_tools::RelativePath path (outputPath, build_tools::RelativePath::buildTargetFolder);

541 void createBuildTarget (XmlElement& xml, CodeBlocksTarget& target, const BuildConfiguration& config) const
542 {
543 xml.setAttribute ("title", target.getTargetNameForConfiguration (config));
544
545 {
546 auto* output = xml.createNewChildElement ("Option");
547
548 output->setAttribute ("output", getOutputPathForTarget (target, config));
549
550 if (isLinux())
551 {
552 bool keepPrefix = (target.type == build_tools::ProjectType::Target::VSTPlugIn);
553
554 output->setAttribute ("prefix_auto", keepPrefix ? 0 : 1);
555 }
556 else
557 {
558 output->setAttribute ("prefix_auto", 0);
559 }
560
561 output->setAttribute ("extension_auto", 0);
562 }
563
564 xml.createNewChildElement ("Option")
565 ->setAttribute ("object_output", "obj/" + File::createLegalFileName (config.getName().trim()));
566
567 xml.createNewChildElement ("Option")->setAttribute ("type", getTypeIndex (target.type));
568 xml.createNewChildElement ("Option")->setAttribute ("compiler", "gcc");
569
570 if (getProject().isAudioPluginProject() && target.type != build_tools::ProjectType::Target::SharedCodeTarget)
571 xml.createNewChildElement ("Option")->setAttribute ("external_deps", getSharedCodePath (config));
572
573 {
574 auto* compiler = xml.createNewChildElement ("Compiler");
575
576 {
577 StringArray flags;
578
579 for (auto& def : getDefines (config, target))
580 {
581 if (! def.containsChar ('='))
582 def << '=';
583
584 flags.add ("-D" + def);
585 }
586
587 flags.addArray (getCompilerFlags (config, target));
588
589 for (auto flag : flags)
590 setAddOption (*compiler, "option", flag);
591 }
592
593 {
594 auto includePaths = getIncludePaths (config);
595
596 for (auto path : includePaths)
597 setAddOption (*compiler, "directory", path);
598 }
599 }
600
601 {
602 auto* linker = xml.createNewChildElement ("Linker");
603
604 if (getProject().isAudioPluginProject() && target.type != build_tools::ProjectType::Target::SharedCodeTarget)
605 setAddOption (*linker, "option", getSharedCodePath (config).quoted());
606
607 for (auto& flag : getLinkerFlags (config, target))
608 setAddOption (*linker, "option", flag);
609
610 const StringArray& libs = isWindows() ? mingwLibs : linuxLibs;
611
612 for (auto lib : libs)
613 setAddOption (*linker, "library", lib);
614
615 for (auto& path : getLinkerSearchPaths (config, target))
616 setAddOption (*linker, "directory",
617 build_tools::replacePreprocessorDefs (getAllPreprocessorDefs(), path));
618 }
619 }

742 void addCompileUnits (const Project::Item& projectItem, XmlElement& xml) const
743 {
744 if (projectItem.isGroup())
745 {
746 for (int i = 0; i < projectItem.getNumChildren(); ++i)
747 addCompileUnits (projectItem.getChild(i), xml);
748 }
749 else if (projectItem.shouldBeAddedToTargetProject() && projectItem.shouldBeAddedToTargetExporter (*this))
750 {
751 build_tools::RelativePath file (projectItem.getFile(), getTargetFolder(), build_tools::RelativePath::buildTargetFolder);
752
753 auto* unit = xml.createNewChildElement ("Unit");
754 unit->setAttribute ("filename", file.toUnixStyle());
755
756 for (ConstConfigIterator config (*this); config.next();)
757 {
758 auto targetName = getTargetForProjectItem (projectItem).getTargetNameForConfiguration (*config);
759 unit->createNewChildElement ("Option")->setAttribute ("target", targetName);
760 }
761
762 if (projectItem.shouldBeCompiled())
763 {
764 auto extraCompilerFlags = getCompilerFlagsForProjectItem (projectItem);
765
766 if (extraCompilerFlags.isNotEmpty())
767 {
768 auto* optionElement = unit->createNewChildElement ("Option");
769
770 optionElement->setAttribute ("compiler", "gcc");
771 optionElement->setAttribute ("use", 1);
772 optionElement->setAttribute ("buildCommand", "$compiler $options " + extraCompilerFlags + " $includes -c $file -o $object");
773 }
774 }
775 else
776 {
777 unit->createNewChildElement ("Option")->setAttribute ("compile", 0);
778 unit->createNewChildElement ("Option")->setAttribute ("link", 0);
779 }
780 }
