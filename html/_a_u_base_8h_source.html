162 // Override this method if your AU processes multiple output busses completely independently --
163 // you'll want to just call Render without the NeedsToRender check.

166 // N.B. Implementations of this method can assume that the output's buffer list has already been
167 // prepared and access it with GetOutput(inBusNumber)->GetBufferList() instead of
168 // GetOutput(inBusNumber)->PrepareBuffer(nFrames) -- if PrepareBuffer is called, a
169 // copy may occur after rendering.
170 virtual OSStatus RenderBus(AudioUnitRenderActionFlags& ioActionFlags,
171 const AudioTimeStamp& inTimeStamp, UInt32 /*inBusNumber*/, UInt32 inNumberFrames)

535 // shared between Render and RenderSlice, inlined to minimize function call overhead
536 OSStatus DoRenderBus(AudioUnitRenderActionFlags& ioActionFlags,
537 const AudioTimeStamp& inTimeStamp, UInt32 inBusNumber, AUOutputElement& theOutput,
538 UInt32 inNumberFrames, AudioBufferList& ioData)
539 {
540 if (ioData.mBuffers[0].mData == nullptr ||
541 (theOutput.WillAllocateBuffer() && Outputs().GetNumberOfElements() > 1)) {
542 // will render into cache buffer
543 theOutput.PrepareBuffer(inNumberFrames);
544 } else {
545 // will render into caller's buffer
546 theOutput.SetBufferList(ioData);
547 }
548 const OSStatus result = RenderBus(ioActionFlags, inTimeStamp, inBusNumber, inNumberFrames);
549 if (result == noErr) {
550 if (ioData.mBuffers[0].mData == nullptr) {
551 theOutput.CopyBufferListTo(ioData);
552 } else {
553 theOutput.CopyBufferContentsTo(ioData);
554 theOutput.InvalidateBufferList();
555 }
556 }
557 return result;
558 }

591 // Usually, you won't override this method. You only need to call this if your DSP code
592 // is prepared to handle scheduled immediate and ramped parameter changes.
593 // Before calling this method, it is assumed you have already called PullInput() on the input
594 // busses for which the DSP code depends. ProcessForScheduledParams() will call (potentially
595 // repeatedly) virtual method ProcessScheduledSlice() to perform the actual DSP for a given
596 // sub-division of the buffer. The job of ProcessForScheduledParams() is to sub-divide the
597 // buffer into smaller pieces according to the scheduled times found in the ParameterEventList
598 // (usually coming directly from a previous call to ScheduleParameter() ), setting the
599 // appropriate immediate or ramped parameter values for the corresponding scopes and elements,
600 // then calling ProcessScheduledSlice() to do the actual DSP for each of these divisions.
