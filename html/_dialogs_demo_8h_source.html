52class DemoBackgroundThread : public ThreadWithProgressWindow
53{
54public:
55 DemoBackgroundThread()
56 : ThreadWithProgressWindow ("busy doing some important things...", true, true)
57 {
58 setStatusMessage ("Getting ready...");
59 }
60
61 void run() override
62 {
63 setProgress (-1.0); // setting a value beyond the range 0 -> 1 will show a spinning bar..
64 setStatusMessage ("Preparing to do some stuff...");
65 wait (2000);
66
67 int thingsToDo = 10;
68
69 for (int i = 0; i < thingsToDo; ++i)
70 {
71 // must check this as often as possible, because this is
72 // how we know if the user's pressed 'cancel'
73 if (threadShouldExit())
74 return;
75
76 // this will update the progress bar on the dialog box
77 setProgress (i / (double) thingsToDo);
78
79 setStatusMessage (String (thingsToDo - i) + " things left to do...");
80
81 wait (500);
82 }
83
84 setProgress (-1.0); // setting a value beyond the range 0 -> 1 will show a spinning bar..
85 setStatusMessage ("Finishing off the last few bits and pieces!");
86 wait (2000);
87 }
88
89 // This method gets called on the message thread once our thread has finished..
90 void threadComplete (bool userPressedCancel) override
91 {
92 const String messageString (userPressedCancel ? "You pressed cancel!" : "Thread finished ok!");
93
94 AlertWindow::showAsync (MessageBoxOptions()
95 .withIconType (MessageBoxIconType::InfoIcon)
96 .withTitle ("Progress window")
97 .withMessage (messageString)
98 .withButton ("OK"),
99 nullptr);
100
101 // ..and clean up by deleting our thread object..
102 delete this;
103 }
104};
