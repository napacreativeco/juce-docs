140template <typename ElementType>
141class ElementListener : private ChangeListener
142{
143public:
144 ElementListener (ElementType* e)
145 : owner (e), broadcaster (*owner->getDocument()),
146 propToRefresh (nullptr)
147 {
148 broadcaster.addChangeListener (this);
149 }
150
151 ~ElementListener() override
152 {
153 jassert (propToRefresh != nullptr);
154 broadcaster.removeChangeListener (this);
155 }
156
157 void setPropertyToRefresh (PropertyComponent& pc)
158 {
159 propToRefresh = &pc;
160 }
161
162 mutable Component::SafePointer<ElementType> owner;
163 ChangeBroadcaster& broadcaster;
164 PropertyComponent* propToRefresh;
165
166private:
167 void changeListenerCallback (ChangeBroadcaster*) override
168 {
169 jassert (propToRefresh != nullptr);
170 if (propToRefresh != nullptr && owner != nullptr)
171 propToRefresh->refresh();
172 }

174 JUCE_DECLARE_NON_COPYABLE (ElementListener)
