273 template <typename Integral, std::enable_if_t<std::is_integral_v<Integral>, int> = 0>
274 static Range findMinAndMax (const ValueType* values, Integral numValues) noexcept
275 {
276 if (numValues <= 0)
277 return Range();
278
279 const ValueType first (*values++);
280 Range r (first, first);
281
282 while (--numValues > 0) // (> 0 rather than >= 0 because we've already taken the first sample)
283 {
284 const ValueType v (*values++);
285
286 if (r.end < v) r.end = v;
287 if (v < r.start) r.start = v;
288 }
289
290 return r;
291 }
