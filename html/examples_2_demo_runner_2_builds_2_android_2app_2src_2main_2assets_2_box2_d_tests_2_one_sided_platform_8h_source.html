77 void PreSolve(b2Contact* contact, const b2Manifold* oldManifold)
78 {
79 Test::PreSolve(contact, oldManifold);
80
81 b2Fixture* fixtureA = contact->GetFixtureA();
82 b2Fixture* fixtureB = contact->GetFixtureB();
83
84 if (fixtureA != m_platform && fixtureA != m_character)
85 {
86 return;
87 }
88
89 if (fixtureB != m_platform && fixtureB != m_character)
90 {
91 return;
92 }
93
94 b2Vec2 position = m_character->GetBody()->GetPosition();
95
96 if (position.y < m_top + m_radius - 3.0f * b2_linearSlop)
97 {
98 contact->SetEnabled(false);
99 }
100 }
