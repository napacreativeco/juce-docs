44class OpenGLContext::NativeContext
45{
46public:
47 NativeContext (Component& c,
48 const OpenGLPixelFormat& pixFormat,
49 void* contextToShare,
50 bool multisampling,
51 OpenGLVersion version)
52 : component (c), openGLversion (version),
53 useDepthBuffer (pixFormat.depthBufferBits > 0),
54 useMSAA (multisampling)
55 {
56 JUCE_AUTORELEASEPOOL
57 {
58 if (auto* peer = component.getPeer())
59 {
60 auto bounds = peer->getAreaCoveredBy (component);
61
62 view = [[JuceGLView alloc] initWithFrame: convertToCGRect (bounds)];
63 view.opaque = YES;
64 view.hidden = NO;
65 view.backgroundColor = [UIColor blackColor];
66 view.userInteractionEnabled = NO;
67
68 glLayer = (CAEAGLLayer*) [view layer];
69 glLayer.opaque = true;
70
71 updateWindowPosition (bounds);
72
73 [((UIView*) peer->getNativeHandle()) addSubview: view];
74
75 const auto shouldUseES3 = version != defaultGLVersion
76 && [[UIDevice currentDevice].systemVersion floatValue] >= 7.0;
77
78 const auto gotContext = (shouldUseES3 && createContext (kEAGLRenderingAPIOpenGLES3, contextToShare))
79 || createContext (kEAGLRenderingAPIOpenGLES2, contextToShare);
80
81 jassertquiet (gotContext);
82
83 if (context != nil)
84 {
85 // I'd prefer to put this stuff in the initialiseOnRenderThread() call, but doing
86 // so causes mysterious timing-related failures.
87 [EAGLContext setCurrentContext: context.get()];
88 gl::loadFunctions();
89 createGLBuffers();
90 deactivateCurrentContext();
91 }
92 else
93 {
94 jassertfalse;
95 }
96 }
97 else
98 {
99 jassertfalse;
100 }
101 }
102 }
103
104 ~NativeContext()
105 {
106 context.reset();
107 [view removeFromSuperview];
108 [view release];
109 }

111 InitResult initialiseOnRenderThread (OpenGLContext&) { return InitResult::success; }

113 void shutdownOnRenderThread()
114 {
115 JUCE_CHECK_OPENGL_ERROR
116 freeGLBuffers();
117 deactivateCurrentContext();
118 }

120 bool createdOk() const noexcept { return getRawContext() != nullptr; }
121 void* getRawContext() const noexcept { return context.get(); }
122 GLuint getFrameBufferID() const noexcept { return useMSAA ? msaaBufferHandle : frameBufferHandle; }

125 {
126 if (! [EAGLContext setCurrentContext: context.get()])
127 return false;

129 glBindFramebuffer (GL_FRAMEBUFFER, useMSAA ? msaaBufferHandle
130 : frameBufferHandle);
131 return true;
132 }

144 void swapBuffers()
145 {
146 if (useMSAA)
147 {
148 glBindFramebuffer (GL_DRAW_FRAMEBUFFER, frameBufferHandle);
149 glBindFramebuffer (GL_READ_FRAMEBUFFER, msaaBufferHandle);
150
151 if (openGLversion >= openGL3_2)
152 {
153 auto w = roundToInt (lastBounds.getWidth() * glLayer.contentsScale);
154 auto h = roundToInt (lastBounds.getHeight() * glLayer.contentsScale);
155
156 glBlitFramebuffer (0, 0, w, h,
157 0, 0, w, h,
158 GL_COLOR_BUFFER_BIT,
159 GL_NEAREST);
160 }
161 else
162 {
163 ::glResolveMultisampleFramebufferAPPLE();
164 }
165 }
166
167 glBindRenderbuffer (GL_RENDERBUFFER, colorBufferHandle);
168 [context.get() presentRenderbuffer: GL_RENDERBUFFER];
169
170 if (needToRebuildBuffers)
171 {
172 needToRebuildBuffers = false;
173
174 freeGLBuffers();
175 createGLBuffers();
176 makeActive();
177 }
178 }

180 void updateWindowPosition (Rectangle<int> bounds)
181 {
182 view.frame = convertToCGRect (bounds);
183 glLayer.contentsScale = (CGFloat) (Desktop::getInstance().getDisplays().getPrimaryDisplay()->scale
184 / component.getDesktopScaleFactor());
185
186 if (lastBounds != bounds)
187 {
188 lastBounds = bounds;
189 needToRebuildBuffers = true;
190 }
191 }

223 bool createContext (EAGLRenderingAPI type, void* contextToShare)
224 {
225 jassert (context == nil);
226 context.reset ([EAGLContext alloc]);
227
228 if (contextToShare != nullptr)
229 [context.get() initWithAPI: type sharegroup: [(EAGLContext*) contextToShare sharegroup]];
230 else
231 [context.get() initWithAPI: type];

237 void createGLBuffers()
238 {
239 glGenFramebuffers (1, &frameBufferHandle);
240 glGenRenderbuffers (1, &colorBufferHandle);
241
242 glBindFramebuffer (GL_FRAMEBUFFER, frameBufferHandle);
243 glBindRenderbuffer (GL_RENDERBUFFER, colorBufferHandle);
244
245 glFramebufferRenderbuffer (GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, colorBufferHandle);
246
247 [[maybe_unused]] bool ok = [context.get() renderbufferStorage: GL_RENDERBUFFER fromDrawable: glLayer];
248 jassert (ok);
249
250 GLint width, height;
251 glGetRenderbufferParameteriv (GL_RENDERBUFFER, GL_RENDERBUFFER_WIDTH, &width);
252 glGetRenderbufferParameteriv (GL_RENDERBUFFER, GL_RENDERBUFFER_HEIGHT, &height);
253
254 if (useMSAA)
255 {
256 glGenFramebuffers (1, &msaaBufferHandle);
257 glGenRenderbuffers (1, &msaaColorHandle);
258
259 glBindFramebuffer (GL_FRAMEBUFFER, msaaBufferHandle);
260 glBindRenderbuffer (GL_RENDERBUFFER, msaaColorHandle);
261
262 glRenderbufferStorageMultisample (GL_RENDERBUFFER, 4, GL_RGBA8, width, height);
263
264 glFramebufferRenderbuffer (GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, msaaColorHandle);
265 }
266
267 if (useDepthBuffer)
268 {
269 glGenRenderbuffers (1, &depthBufferHandle);
270 glBindRenderbuffer (GL_RENDERBUFFER, depthBufferHandle);
271
272 if (useMSAA)
273 glRenderbufferStorageMultisample (GL_RENDERBUFFER, 4, GL_DEPTH_COMPONENT16, width, height);
274 else
275 glRenderbufferStorage (GL_RENDERBUFFER, GL_DEPTH_COMPONENT16, width, height);
276
277 glFramebufferRenderbuffer (GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_RENDERBUFFER, depthBufferHandle);
278 }
279
280 jassert (glCheckFramebufferStatus (GL_FRAMEBUFFER) == GL_FRAMEBUFFER_COMPLETE);
281 JUCE_CHECK_OPENGL_ERROR
282 }
283
284 void freeGLBuffers()
285 {
286 JUCE_CHECK_OPENGL_ERROR
287 [context.get() renderbufferStorage: GL_RENDERBUFFER fromDrawable: nil];
288
289 deleteFrameBuffer (frameBufferHandle);
290 deleteFrameBuffer (msaaBufferHandle);
291 deleteRenderBuffer (colorBufferHandle);
292 deleteRenderBuffer (depthBufferHandle);
293 deleteRenderBuffer (msaaColorHandle);
294
295 JUCE_CHECK_OPENGL_ERROR
296 }

298 static void deleteFrameBuffer (GLuint& i) { if (i != 0) glDeleteFramebuffers (1, &i); i = 0; }
299 static void deleteRenderBuffer (GLuint& i) { if (i != 0) glDeleteRenderbuffers (1, &i); i = 0; }
