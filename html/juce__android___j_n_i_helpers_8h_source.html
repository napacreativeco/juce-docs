30template <typename JavaType>
31class LocalRef
32{
33public:
34 LocalRef() noexcept : obj (nullptr) {}
35 explicit LocalRef (JavaType o) noexcept : obj (o) {}
36 LocalRef (const LocalRef& other) noexcept : obj (retain (other.obj)) {}
37 LocalRef (LocalRef&& other) noexcept : obj (nullptr) { std::swap (obj, other.obj); }
38 ~LocalRef() { clear(); }
39
40 void clear()
41 {
42 if (obj != nullptr)
43 {
44 getEnv()->DeleteLocalRef (obj);
45 obj = nullptr;
46 }
47 }
48
49 LocalRef& operator= (const LocalRef& other)
50 {
51 JavaType newObj = retain (other.obj);
52 clear();
53 obj = newObj;
54 return *this;
55 }
56
57 LocalRef& operator= (LocalRef&& other)
58 {
59 clear();
60 std::swap (other.obj, obj);
61 return *this;
62 }
63
64 operator JavaType() const noexcept { return obj; }
65 JavaType get() const noexcept { return obj; }
66
67private:
68 JavaType obj;
69
70 static JavaType retain (JavaType obj)
71 {
72 return obj == nullptr ? nullptr : (JavaType) getEnv()->NewLocalRef (obj);
73 }
74};

77template <typename JavaType>
78class GlobalRefImpl
79{
80public:
81 GlobalRefImpl() noexcept : obj (nullptr) {}
82 explicit GlobalRefImpl (const LocalRef<JavaType>& o) : obj (retain (o.get(), getEnv())) {}
83 GlobalRefImpl (const LocalRef<JavaType>& o, JNIEnv* env) : obj (retain (o.get(), env)) {}
84 GlobalRefImpl (const GlobalRefImpl& other) : obj (retain (other.obj, getEnv())) {}
85 GlobalRefImpl (GlobalRefImpl&& other) noexcept : obj (nullptr) { std::swap (other.obj, obj); }
86 ~GlobalRefImpl() { clear(); }
87
88 void clear() { if (obj != nullptr) clear (getEnv()); }
89 void clear (JNIEnv* env)
90 {
91 if (obj != nullptr)
92 {
93 env->DeleteGlobalRef (obj);
94 obj = nullptr;
95 }
96 }
97
98 GlobalRefImpl& operator= (const GlobalRefImpl& other)
99 {
100 JavaType newObj = retain (other.obj, getEnv());
101 clear();
102 obj = newObj;
103 return *this;
104 }
105
106 GlobalRefImpl& operator= (GlobalRefImpl&& other)
107 {
108 clear();
109 std::swap (obj, other.obj);
110
111 return *this;
112 }

119 #define DECLARE_CALL_TYPE_METHOD(returnType, typeName) \
120 returnType call##typeName##Method (jmethodID methodID, ... ) const \
121 { \
122 va_list args; \
123 va_start (args, methodID); \
124 returnType result = getEnv()->Call##typeName##MethodV (obj, methodID, args); \
125 va_end (args); \
126 return result; \
127 }
128
129 DECLARE_CALL_TYPE_METHOD (jobject, Object)
130 DECLARE_CALL_TYPE_METHOD (jboolean, Boolean)
131 DECLARE_CALL_TYPE_METHOD (jbyte, Byte)
132 DECLARE_CALL_TYPE_METHOD (jchar, Char)
133 DECLARE_CALL_TYPE_METHOD (jshort, Short)
134 DECLARE_CALL_TYPE_METHOD (jint, Int)
135 DECLARE_CALL_TYPE_METHOD (jlong, Long)
136 DECLARE_CALL_TYPE_METHOD (jfloat, Float)
137 DECLARE_CALL_TYPE_METHOD (jdouble, Double)
138 #undef DECLARE_CALL_TYPE_METHOD
139
140 void callVoidMethod (jmethodID methodID, ... ) const
141 {
142 va_list args;
143 va_start (args, methodID);
144 getEnv()->CallVoidMethodV (obj, methodID, args);
145 va_end (args);
146 }
147
148private:
149 //==============================================================================
150 JavaType obj = nullptr;
151
152 static JavaType retain (JavaType obj, JNIEnv* env)
153 {
154 return obj != nullptr ? static_cast<JavaType> (env->NewGlobalRef (obj))
155 : nullptr;
156 }
157};

225#define DECLARE_JNI_CLASS_WITH_BYTECODE(CppClassName, javaPath, minSDK, byteCodeData) \
226 class CppClassName ## _Class : public JNIClassBase \
227 { \
228 public: \
229 CppClassName ## _Class() : JNIClassBase (javaPath, minSDK, byteCodeData, numBytes (byteCodeData)) {} \
230 \
231 void initialiseFields (JNIEnv* env) \
232 { \
233 Array<JNINativeMethod> callbacks; \
234 JNI_CLASS_MEMBERS (CREATE_JNI_METHOD, CREATE_JNI_STATICMETHOD, CREATE_JNI_FIELD, CREATE_JNI_STATICFIELD, CREATE_JNI_CALLBACK); \
235 resolveCallbacks (env, callbacks); \
236 } \
237 \
238 JNI_CLASS_MEMBERS (DECLARE_JNI_METHOD, DECLARE_JNI_METHOD, DECLARE_JNI_FIELD, DECLARE_JNI_FIELD, DECLARE_JNI_CALLBACK) \
239 }; \
240 static inline const CppClassName ## _Class CppClassName;

802 inline String juceString (JNIEnv* env, jstring s)
803 {
804 if (s == nullptr)
805 return {};
806
807 const char* const utf8 = env->GetStringUTFChars (s, nullptr);
808 CharPointer_UTF8 utf8CP (utf8);
809 const String result (utf8CP);
810 env->ReleaseStringUTFChars (s, utf8);
811 return result;
812 }
813
814 inline String juceString (jstring s)
815 {
816 return juceString (getEnv(), s);
817 }
818
819 inline LocalRef<jstring> javaString (const String& s)
820 {
821 return LocalRef<jstring> (getEnv()->NewStringUTF (s.toUTF8()));
822 }
823
824 inline LocalRef<jstring> javaStringFromChar (const juce_wchar c)
825 {
826 char utf8[8] = { 0 };
827 CharPointer_UTF8 (utf8).write (c);
828 return LocalRef<jstring> (getEnv()->NewStringUTF (utf8));
829 }
830
831 inline LocalRef<jobjectArray> juceStringArrayToJava (const StringArray& juceArray)
832 {
833 auto* env = getEnv();
834
835 LocalRef<jobjectArray> result (env->NewObjectArray ((jsize) juceArray.size(),
836 JavaString,
837 javaString ("").get()));
838
839 for (int i = 0; i < juceArray.size(); ++i)
840 env->SetObjectArrayElement (result, i, javaString (juceArray [i]).get());
841
842 return result;
843 }
844
845 inline StringArray javaStringArrayToJuce (const LocalRef<jobjectArray>& javaArray)
846 {
847 if (javaArray.get() == nullptr)
848 return {};
849
850 auto* env = getEnv();
851
852 StringArray result;
853
854 for (int i = 0; i < env->GetArrayLength (javaArray.get()); ++i)
855 {
856 LocalRef<jstring> javaString ((jstring) env->GetObjectArrayElement (javaArray.get(), i));
857 result.add (juceString (javaString.get()));
858 }
859
860 return result;
861 }

863 inline bool jniCheckHasExceptionOccurredAndClear()
864 {
865 auto* env = getEnv();
866
867 const auto result = env->ExceptionCheck();
868 #if JUCE_DEBUG
869 env->ExceptionDescribe();
870 #endif
871 env->ExceptionClear();
872 return result;
873 }

951struct SurfaceHolderCallback : AndroidInterfaceImplementer
952{
953 virtual ~SurfaceHolderCallback() override = default;
954
955 virtual void surfaceChanged (LocalRef<jobject> holder, int format, int width, int height) = 0;
956 virtual void surfaceCreated (LocalRef<jobject> holder) = 0;
957 virtual void surfaceDestroyed (LocalRef<jobject> holder) = 0;
958
959private:
960 jobject invoke (jobject proxy, jobject method, jobjectArray args) override
961 {
962 auto* env = getEnv();
963 auto methodName = juceString ((jstring) env->CallObjectMethod (method, JavaMethod.getName));
964 LocalRef<jobject> holder (env->GetArrayLength (args) > 0 ? env->GetObjectArrayElement (args, 0) : (jobject) nullptr);
965
966 if (methodName == "surfaceChanged")
967 {
968 int intArgs[3];
969
970 for (int i = 0; i < 3; ++i)
971 {
972 LocalRef<jobject> boxedType (env->GetObjectArrayElement (args, 1 + i));
973 intArgs[i] = env->CallIntMethod (boxedType.get(), JavaInteger.intValue);
974 }
975
976 surfaceChanged (std::move (holder), intArgs[0], intArgs[1], intArgs[2]);
977 }
978 else if (methodName == "surfaceCreated")
979 {
980 surfaceCreated (std::move (holder));
981 }
982 else if (methodName == "surfaceDestroyed")
983 {
984 surfaceDestroyed (std::move (holder));
985 }
986 else
987 {
988 return AndroidInterfaceImplementer::invoke (proxy, method, args);
989 }
990
991 return nullptr;
992 }
993};

1039template <auto Fn, typename Result, typename Class, typename... Args>
1040inline constexpr auto generatedCallbackImpl =
1041 juce::toFnPtr (JNICALL [] (JNIEnv* env, jobject, jlong host, Args... args) -> Result
1042 {
1043 if (auto* object = reinterpret_cast<Class*> (host))
1044 return Fn (env, *object, args...);
1045
1046 return {};
1047 });
1048
1049template <auto Fn, typename Result, typename Class, typename... Args>
1050constexpr auto generateCallbackImpl (Result (*) (JNIEnv*, Class&, Args...)) { return generatedCallbackImpl<Fn, Result, Class, Args...>; }
1051
1052template <auto Fn, typename Result, typename Class, typename... Args>
1053constexpr auto generateCallbackImpl (Result (*) (JNIEnv*, const Class&, Args...)) { return generatedCallbackImpl<Fn, Result, Class, Args...>; }
1054
1055} // namespace detail
