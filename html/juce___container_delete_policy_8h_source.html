39template <typename ObjectType>
40struct ContainerDeletePolicy
41{
42 static void destroy (ObjectType* object)
43 {
44 // If the line below triggers a compiler error, it means that you are using
45 // an incomplete type for ObjectType (for example, a type that is declared
46 // but not defined). This is a problem because then the following delete is
47 // undefined behaviour. The purpose of the sizeof is to capture this situation.
48 // If this was caused by a OwnedArray of a forward-declared type, move the
49 // implementation of all methods trying to use the OwnedArray (e.g. the destructor
50 // of the class owning it) into cpp files where they can see to the definition
51 // of ObjectType. This should fix the error.
52 [[maybe_unused]] constexpr auto size = sizeof (ObjectType);
53
54 delete object;
55 }
56};
