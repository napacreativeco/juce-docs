77 static std::vector<NSOpenGLPixelFormatAttribute> createAttribs (OpenGLVersion version,
78 const OpenGLPixelFormat& pixFormat,
79 bool shouldUseMultisampling)
80 {
81 std::vector<NSOpenGLPixelFormatAttribute> attribs
82 {
83 NSOpenGLPFAOpenGLProfile, [version]
84 {
85 if (version == openGL3_2)
86 return NSOpenGLProfileVersion3_2Core;
87
88 if (version != defaultGLVersion)
89 if (@available (macOS 10.10, *))
90 return NSOpenGLProfileVersion4_1Core;
91
92 return NSOpenGLProfileVersionLegacy;
93 }(),
94 NSOpenGLPFADoubleBuffer,
95 NSOpenGLPFAClosestPolicy,
96 NSOpenGLPFANoRecovery,
97 NSOpenGLPFAColorSize, static_cast<NSOpenGLPixelFormatAttribute> (pixFormat.redBits + pixFormat.greenBits + pixFormat.blueBits),
98 NSOpenGLPFAAlphaSize, static_cast<NSOpenGLPixelFormatAttribute> (pixFormat.alphaBits),
99 NSOpenGLPFADepthSize, static_cast<NSOpenGLPixelFormatAttribute> (pixFormat.depthBufferBits),
100 NSOpenGLPFAStencilSize, static_cast<NSOpenGLPixelFormatAttribute> (pixFormat.stencilBufferBits),
101 NSOpenGLPFAAccumSize, static_cast<NSOpenGLPixelFormatAttribute> (pixFormat.accumulationBufferRedBits + pixFormat.accumulationBufferGreenBits
102 + pixFormat.accumulationBufferBlueBits + pixFormat.accumulationBufferAlphaBits)
103 };
104
105 if (shouldUseMultisampling)
106 {
107 attribs.insert (attribs.cend(),
108 {
109 NSOpenGLPFAMultisample,
110 NSOpenGLPFASampleBuffers, static_cast<NSOpenGLPixelFormatAttribute> (1),
111 NSOpenGLPFASamples, static_cast<NSOpenGLPixelFormatAttribute> (pixFormat.multisamplingLevel)
112 });
113 }
114
115 attribs.push_back (0);
116
117 return attribs;
118 }

170 void swapBuffers()
171 {
172 auto now = Time::getMillisecondCounterHiRes();
173 [renderContext flushBuffer];
174
175 if (minSwapTimeMs > 0)
176 {
177 // When our window is entirely occluded by other windows, flushBuffer
178 // fails to wait for the swap interval, so the render loop spins at full
179 // speed, burning CPU. This hack detects when things are going too fast
180 // and sleeps if necessary.
181
182 auto swapTime = Time::getMillisecondCounterHiRes() - now;
183 auto frameTime = (int) (now - lastSwapTime);
184
185 if (swapTime < 0.5 && frameTime < minSwapTimeMs - 3)
186 {
187 if (underrunCounter > 3)
188 {
189 Thread::sleep (2 * (minSwapTimeMs - frameTime));
190 now = Time::getMillisecondCounterHiRes();
191 }
192 else
193 {
194 ++underrunCounter;
195 }
196 }
197 else
198 {
199 if (underrunCounter > 0)
200 --underrunCounter;
201 }
202 }
203
204 lastSwapTime = now;
205 }

218 bool setSwapInterval (int numFramesPerSwapIn)
219 {
220 numFramesPerSwap = numFramesPerSwapIn;
221
222 // The macOS OpenGL programming guide says that numFramesPerSwap
223 // can only be 0 or 1.
224 jassert (isPositiveAndBelow (numFramesPerSwap, 2));
225
226 [renderContext setValues: (const GLint*) &numFramesPerSwap
227 forParameter: getSwapIntervalParameter()];
228
229 updateMinSwapTime();
230
231 return true;
232 }
