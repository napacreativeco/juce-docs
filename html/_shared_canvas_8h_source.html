125 void save (OutputStream& out) const
126 {
127 out.writeInt (magic);
128 out.writeInt ((int) backgroundColour.getARGB());
129
130 out.writeInt (clients.size());
131
132 for (const auto& c : clients)
133 {
134 out.writeString (c.name);
135 writePoint (out, c.centre);
136 out.writeFloat (c.scaleFactor);
137 }
138
139 out.writeInt (paths.size());
140
141 for (const auto& p : paths)
142 {
143 writeFill (out, p.fill);
144 p.path.writePathToStream (out);
145 }
146 }
147
148 void load (InputStream& in)
149 {
150 if (in.readInt() != magic)
151 return;
152
153 backgroundColour = Colour ((uint32) in.readInt());
154
155 {
156 const int numClients = in.readInt();
157 clients.clearQuick();
158
159 for (int i = 0; i < numClients; ++i)
160 {
161 ClientArea c;
162 c.name = in.readString();
163 c.centre = readPoint (in);
164 c.scaleFactor = in.readFloat();
165 clients.add (c);
166 }
167 }
168
169 {
170 const int numPaths = in.readInt();
171 paths.clearQuick();
172
173 for (int i = 0; i < numPaths; ++i)
174 {
175 ColouredPath p;
176 p.fill = readFill (in);
177 p.path.loadPathFromStream (in);
178
179 paths.add (std::move (p));
180 }
181 }
182 }

184 MemoryBlock toMemoryBlock() const
185 {

187 save (o);

223 static void writeFill (OutputStream& out, const FillType& f)
224 {
225 if (f.isColour())
226 {
227 out.writeByte (0);
228 out.writeInt ((int) f.colour.getARGB());
229 }
230 else if (f.isGradient())
231 {
232 const ColourGradient& cg = *f.gradient;
233 jassert (cg.getNumColours() >= 2);
234
235 out.writeByte (cg.isRadial ? 2 : 1);
236
237 writePoint (out, cg.point1);
238 writePoint (out, cg.point2);
239
240 out.writeCompressedInt (cg.getNumColours());
241
242 for (int i = 0; i < cg.getNumColours(); ++i)
243 {
244 out.writeDouble (cg.getColourPosition (i));
245 out.writeInt ((int) cg.getColour(i).getARGB());
246 }
247 }
248 else
249 {
250 jassertfalse;
251 }
252 }
253
254 static FillType readFill (InputStream& in)
255 {
256 int type = in.readByte();
257
258 if (type == 0)
259 return FillType (Colour ((uint32) in.readInt()));
260
261 if (type > 2)
262 {
263 jassertfalse;
264 return FillType();
265 }
266
267 ColourGradient cg;
268 cg.point1 = readPoint (in);
269 cg.point2 = readPoint (in);
270
271 cg.clearColours();
272
273 int numColours = in.readCompressedInt();
274
275 for (int i = 0; i < numColours; ++i)
276 {
277 const double pos = in.readDouble();
278 cg.addColour (pos, Colour ((uint32) in.readInt()));
279 }
280
281 jassert (cg.getNumColours() >= 2);
282
283 return FillType (cg);
284 }

446struct BlockPacketiser
447{
448 void createBlocksFromData (const MemoryBlock& data, size_t maxBlockSize)
449 {
450 jassert (blocks.size() == 0);
451
452 int offset = 0;
453 size_t remaining = data.getSize();
454
455 while (remaining > 0)
456 {
457 auto num = (size_t) jmin (maxBlockSize, remaining);
458 blocks.add (MemoryBlock (addBytesToPointer (data.getData(), offset), num));
459 offset += (int) num;
460 remaining -= num;
461 }
462
463 MemoryOutputStream checksumBlock;
464 checksumBlock << getLastPacketPrefix() << MD5 (data).toHexString() << (char) 0 << (char) 0;
465 blocks.add (checksumBlock.getMemoryBlock());
466
467 for (int i = 0; i < blocks.size(); ++i)
468 {
469 auto index = (uint32) ByteOrder::swapIfBigEndian (i);
470 blocks.getReference(i).append (&index, sizeof (index));
471 }
472 }
473
474 // returns true if this is an end-of-sequence block
475 bool appendIncomingBlock (MemoryBlock data)
476 {
477 if (data.getSize() > 4)
478 blocks.addSorted (*this, data);
479
480 return String (CharPointer_ASCII ((const char*) data.getData())).startsWith (getLastPacketPrefix());
481 }
482
483 bool reassemble (MemoryBlock& result)
484 {
485 result.reset();
486
487 if (blocks.size() > 1)
488 {
489 for (int i = 0; i < blocks.size() - 1; ++i)
490 result.append (blocks.getReference(i).getData(), blocks.getReference(i).getSize() - 4);
491
492 String storedMD5 (String (CharPointer_ASCII ((const char*) blocks.getLast().getData()))
493 .fromFirstOccurrenceOf (getLastPacketPrefix(), false, false));
494
495 blocks.clearQuick();
496
497 if (MD5 (result).toHexString().trim().equalsIgnoreCase (storedMD5.trim()))
498 return true;
499 }
500
501 result.reset();
502 return false;
503 }
504
505 static int compareElements (const MemoryBlock& b1, const MemoryBlock& b2)
506 {
507 auto i1 = ByteOrder::littleEndianInt (addBytesToPointer (b1.getData(), b1.getSize() - 4));
508 auto i2 = ByteOrder::littleEndianInt (addBytesToPointer (b2.getData(), b2.getSize() - 4));
509 return (int) (i1 - i2);
510 }
511
512 static const char* getLastPacketPrefix() { return "**END_OF_PACKET_LIST** "; }
513
514 Array<MemoryBlock> blocks;
