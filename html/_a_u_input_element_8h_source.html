57inline OSStatus AUInputElement::PullInputWithBufferList(AudioUnitRenderActionFlags& ioActionFlags,
58 const AudioTimeStamp& inTimeStamp, AudioUnitElement inElement, UInt32 nFrames,
59 AudioBufferList& inBufferList)
60{
61 OSStatus theResult = noErr;
62
63 if (HasConnection()) {
64 // only support connections for V2 audio units
65 theResult = AudioUnitRender(mConnection.sourceAudioUnit, &ioActionFlags, &inTimeStamp,
66 mConnection.sourceOutputNumber, nFrames, &inBufferList);
67 } else {
68 // kFromCallback:
69 theResult = (mInputProc)(mInputProcRefCon, &ioActionFlags, &inTimeStamp, inElement, nFrames,
70 &inBufferList);
71 }
72
73 if (mInputType == EInputType::NoInput) { // defense: the guy upstream could have disconnected
74 // it's a horrible thing to do, but may happen!
75 return kAudioUnitErr_NoConnection;
76 }
77
78#if !TARGET_OS_IPHONE || DEBUG
79 if (theResult == noErr) { // if there's already an error, there's no point (and maybe some harm)

82 return kAudioUnitErr_InvalidPropertyValue;
