96struct BallComponent : public Component
97{
98 BallComponent (Point<float> pos)
99 : position (pos),
100 speed (Random::getSystemRandom().nextFloat() * 4.0f - 2.0f,
101 Random::getSystemRandom().nextFloat() * -6.0f - 2.0f),
102 colour (Colours::white)
103 {
104 setSize (20, 20);
105 step();
106 }
107
108 bool step()
109 {
110 position += speed;
111 speed.y += 0.1f;
112
113 setCentrePosition ((int) position.x,
114 (int) position.y);
115
116 if (auto* parent = getParentComponent())
117 return isPositiveAndBelow (position.x, (float) parent->getWidth())
118 && position.y < (float) parent->getHeight();
119
120 return position.y < 400.0f && position.x >= -10.0f;
121 }
122
123 void paint (Graphics& g) override
124 {
125 g.setColour (colour);
126 g.fillEllipse (2.0f, 2.0f, (float) getWidth() - 4.0f, (float) getHeight() - 4.0f);
127
128 g.setColour (Colours::darkgrey);
129 g.drawEllipse (2.0f, 2.0f, (float) getWidth() - 4.0f, (float) getHeight() - 4.0f, 1.0f);
130 }
131
132 Point<float> position, speed;
133 Colour colour;
134
135 JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR (BallComponent)
136};

253 void triggerAnimation()
254 {
255 auto width = getWidth();
256 auto height = getHeight();
257
258 bool useWidth = (height > width);
259
260 for (auto* component : componentsToAnimate)
261 {
262 auto newIndex = (componentsToAnimate.indexOf (component) + 3 * cycleCount)
263 % componentsToAnimate.size();
264
265 auto angle = (float) newIndex * MathConstants<float>::twoPi / (float) componentsToAnimate.size();
266
267 auto radius = useWidth ? (float) width * 0.35f
268 : (float) height * 0.35f;
269
270 Rectangle<int> r (getWidth() / 2 + (int) (radius * std::sin (angle)) - 50,
271 getHeight() / 2 + (int) (radius * std::cos (angle)) - 50,
272 100, 100);
273
274 animator.animateComponent (component, r.reduced (10), 1.0f,
275 900 + (int) (300 * std::sin (angle)),
276 false, 0.0, 0.0);
277 }
278
279 ++cycleCount;
280 }

282 void timerCallback() override
283 {
284 if (firstCallback)
285 {
286 triggerAnimation();
287 firstCallback = false;
288 }
289
290 // Go through each of our balls and update their position
291 for (int i = balls.size(); --i >= 0;)
292 if (! balls.getUnchecked (i)->step())
293 balls.remove (i);
294
295 // Randomly generate new balls
296 if (Random::getSystemRandom().nextInt (100) < 4)
297 addAndMakeVisible (balls.add (new BallComponent (ballGenerator.getBounds().getCentre().toFloat())));
298 }
