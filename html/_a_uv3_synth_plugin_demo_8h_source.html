106 //==============================================================================
107 void drawLinearSlider (Graphics& g, int x, int y, int width, int height,
108 float sliderPos, float minSliderPos, float maxSliderPos,
109 const Slider::SliderStyle style, Slider& slider) override
110 {
111 ignoreUnused (style, minSliderPos, maxSliderPos);
112
113 auto r = Rectangle<int> (x + haloRadius, y, width - (haloRadius * 2), height);
114 auto backgroundBar = r.withSizeKeepingCentre(r.getWidth(), 2);
115
116 sliderPos = (sliderPos - minSliderPos) / static_cast<float> (width);
117
118 auto knobPos = static_cast<int> (sliderPos * (float) r.getWidth());
119
120 g.setColour (sliderActivePart);
121 g.fillRect (backgroundBar.removeFromLeft (knobPos));
122
123 g.setColour (sliderInactivePart);
124 g.fillRect (backgroundBar);
125
126 if (slider.isMouseOverOrDragging())
127 {
128 auto haloBounds = r.withTrimmedLeft (knobPos - haloRadius)
129 .withWidth (haloRadius * 2)
130 .withSizeKeepingCentre (haloRadius * 2, haloRadius * 2);
131
132 g.setColour (sliderActivePart.withAlpha (0.5f));
133 g.fillEllipse (haloBounds.toFloat());
134 }
135
136 auto knobRadius = slider.isMouseOverOrDragging() ? knobActiveRadius : knobInActiveRadius;
137 auto knobBounds = r.withTrimmedLeft (knobPos - knobRadius)
138 .withWidth (knobRadius * 2)
139 .withSizeKeepingCentre (knobRadius * 2, knobRadius * 2);
140
141 g.setColour (sliderActivePart);
142 g.fillEllipse (knobBounds.toFloat());
143 }

312class AUv3SynthProcessor : public AudioProcessor
313{
314public:
315 AUv3SynthProcessor ()
316 : AudioProcessor (BusesProperties().withOutput ("Output", AudioChannelSet::stereo(), true)),
317 currentRecording (1, 1), currentProgram (0)
318 {
319 // initialize parameters
320 addParameter (isRecordingParam = new AudioParameterBool ({ "isRecording", 1 }, "Is Recording", false));
321 addParameter (roomSizeParam = new AudioParameterFloat ({ "roomSize", 1 }, "Room Size", 0.0f, 1.0f, 0.5f));
322
323 formatManager.registerBasicFormats();
324
325 for (auto i = 0; i < maxNumVoices; ++i)
326 synth.addVoice (new SamplerVoice());
327
328 loadNewSample (createAssetInputStream ("singing.ogg"), "ogg");
329 }
330
331 //==============================================================================
332 bool isBusesLayoutSupported (const BusesLayout& layouts) const override
333 {
334 return (layouts.getMainOutputChannels() <= 2);
335 }
336
337 void prepareToPlay (double sampleRate, int estimatedMaxSizeOfBuffer) override
338 {
339 ignoreUnused (estimatedMaxSizeOfBuffer);
340
341 lastSampleRate = sampleRate;
342
343 currentRecording.setSize (1, static_cast<int> (std::ceil (maxDurationOfRecording * lastSampleRate)));
344 samplesRecorded = 0;
345
346 synth.setCurrentPlaybackSampleRate (lastSampleRate);
347 reverb.setSampleRate (lastSampleRate);
348 }
349
350 void processBlock (AudioBuffer<float>& buffer, MidiBuffer& midiMessages) override
351 {
352 Reverb::Parameters reverbParameters;
353 reverbParameters.roomSize = roomSizeParam->get();
354
355 reverb.setParameters (reverbParameters);
356 synth.renderNextBlock (buffer, midiMessages, 0, buffer.getNumSamples());
357
358 if (getMainBusNumOutputChannels() == 1)
359 reverb.processMono (buffer.getWritePointer (0), buffer.getNumSamples());
360 else if (getMainBusNumOutputChannels() == 2)
361 reverb.processStereo (buffer.getWritePointer (0), buffer.getWritePointer (1), buffer.getNumSamples());
362 }
363
364 using AudioProcessor::processBlock;
365
366 //==============================================================================
367 void releaseResources() override { currentRecording.setSize (1, 1); }

421 void loadNewSampleBinary (const void* data, int dataSize, const char* format)
422 {
423 auto soundBuffer = std::make_unique<MemoryInputStream> (data, static_cast<std::size_t> (dataSize), false);
424 loadNewSample (std::move (soundBuffer), format);
425 }
426
427 void loadNewSample (std::unique_ptr<InputStream> soundBuffer, const char* format)
428 {
429 std::unique_ptr<AudioFormatReader> formatReader (formatManager.findFormatForFileExtension (format)->createReaderFor (soundBuffer.release(), true));
430
431 BigInteger midiNotes;
432 midiNotes.setRange (0, 126, true);
433 SynthesiserSound::Ptr newSound = new SamplerSound ("Voice", *formatReader, midiNotes, 0x40, 0.0, 0.0, 10.0);
434 synth.removeSound (0);
435 sound = newSound;
436 synth.addSound (sound);
437 }
438
439 void swapSamples()
440 {
441 MemoryBlock mb;
442 auto* stream = new MemoryOutputStream (mb, true);
443
444 {
445 std::unique_ptr<AudioFormatWriter> writer (formatManager.findFormatForFileExtension ("wav")->createWriterFor (stream, lastSampleRate, 1, 16,
446 StringPairArray(), 0));
447 writer->writeFromAudioSampleBuffer (currentRecording, 0, currentRecording.getNumSamples());
448 writer->flush();
449 stream->flush();
450 }
451
452 loadNewSampleBinary (mb.getData(), static_cast<int> (mb.getSize()), "wav");
453 }
