96 NativeContext (Component& comp,
97 const OpenGLPixelFormat& cPixelFormat,
98 void* shareContext,
99 bool useMultisamplingIn,
100 OpenGLVersion)
101 : component (comp), contextToShareWith (shareContext), dummy (*this)
102 {
103 display = XWindowSystem::getInstance()->getDisplay();
104
105 XWindowSystemUtilities::ScopedXLock xLock;
106
107 X11Symbols::getInstance()->xSync (display, False);
108
109 const std::vector<GLint> optionalAttribs
110 {
111 GLX_SAMPLE_BUFFERS, useMultisamplingIn ? 1 : 0,
112 GLX_SAMPLES, cPixelFormat.multisamplingLevel
113 };
114
115 if (! tryChooseVisual (cPixelFormat, optionalAttribs) && ! tryChooseVisual (cPixelFormat, {}))
116 return;
117
118 auto* peer = component.getPeer();
119 jassert (peer != nullptr);
120
121 auto windowH = (Window) peer->getNativeHandle();
122 auto visual = glXGetVisualFromFBConfig (display, *bestConfig);
123 auto colourMap = X11Symbols::getInstance()->xCreateColormap (display, windowH, visual->visual, AllocNone);
124
125 XSetWindowAttributes swa;
126 swa.colormap = colourMap;
127 swa.border_pixel = 0;
128 swa.event_mask = embeddedWindowEventMask;
129
130 auto glBounds = component.getTopLevelComponent()->getLocalArea (&component, component.getLocalBounds());
131
132 glBounds = Desktop::getInstance().getDisplays().logicalToPhysical (glBounds);
133
134 embeddedWindow = X11Symbols::getInstance()->xCreateWindow (display, windowH,
135 glBounds.getX(), glBounds.getY(),
136 (unsigned int) jmax (1, glBounds.getWidth()),
137 (unsigned int) jmax (1, glBounds.getHeight()),
138 0, visual->depth,
139 InputOutput,
140 visual->visual,
141 CWBorderPixel | CWColormap | CWEventMask,
142 &swa);
143
144 peerListener.emplace (component, embeddedWindow);
145
146 X11Symbols::getInstance()->xMapWindow (display, embeddedWindow);
147 X11Symbols::getInstance()->xFreeColormap (display, colourMap);

151 juce_LinuxAddRepaintListener (peer, &dummy);

179 InitResult initialiseOnRenderThread (OpenGLContext& c)
180 {
181 XWindowSystemUtilities::ScopedXLock xLock;
182
183 const auto components = [&]() -> Optional<Version>
184 {
185 switch (c.versionRequired)
186 {
187 case OpenGLVersion::openGL3_2: return Version { 3, 2 };
188 case OpenGLVersion::openGL4_1: return Version { 4, 1 };
189 case OpenGLVersion::openGL4_3: return Version { 4, 3 };
190
191 case OpenGLVersion::defaultGLVersion: break;
192 }
193
194 return {};
195 }();
196
197 if (components.hasValue())
198 {
199 using GLXCreateContextAttribsARB = GLXContext (*) (Display*, GLXFBConfig, GLXContext, Bool, const int*);
200
201 if (const auto glXCreateContextAttribsARB = (GLXCreateContextAttribsARB) OpenGLHelpers::getExtensionFunction ("glXCreateContextAttribsARB"))
202 {
203 #if JUCE_DEBUG
204 constexpr auto contextFlags = GLX_CONTEXT_DEBUG_BIT_ARB;
205 #else
206 constexpr auto contextFlags = 0;
207 #endif
208
209 const int attribs[]
210 {
211 GLX_CONTEXT_MAJOR_VERSION_ARB, components->major,
212 GLX_CONTEXT_MINOR_VERSION_ARB, components->minor,
213 GLX_CONTEXT_PROFILE_MASK_ARB, GLX_CONTEXT_CORE_PROFILE_BIT_ARB,
214 GLX_CONTEXT_FLAGS_ARB, contextFlags,
215 None
216 };
217
218 renderContext = glXCreateContextAttribsARB (display, *bestConfig, (GLXContext) contextToShareWith, GL_TRUE, attribs);
219 }
220 }
221
222 if (renderContext == nullptr)
223 renderContext = glXCreateNewContext (display, *bestConfig, GLX_RGBA_TYPE, (GLXContext) contextToShareWith, GL_TRUE);
224
225 if (renderContext == nullptr)
226 return InitResult::fatal;
227
228 c.makeActive();
229 context = &c;
230 return InitResult::success;
231 }

317 bool tryChooseVisual (const OpenGLPixelFormat& format, const std::vector<GLint>& optionalAttribs)
318 {
319 std::vector<GLint> allAttribs
320 {
321 GLX_RENDER_TYPE, GLX_RGBA_BIT,
322 GLX_DOUBLEBUFFER, True,
323 GLX_RED_SIZE, format.redBits,
324 GLX_GREEN_SIZE, format.greenBits,
325 GLX_BLUE_SIZE, format.blueBits,
326 GLX_ALPHA_SIZE, format.alphaBits,
327 GLX_DEPTH_SIZE, format.depthBufferBits,
328 GLX_STENCIL_SIZE, format.stencilBufferBits,
329 GLX_ACCUM_RED_SIZE, format.accumulationBufferRedBits,
330 GLX_ACCUM_GREEN_SIZE, format.accumulationBufferGreenBits,
331 GLX_ACCUM_BLUE_SIZE, format.accumulationBufferBlueBits,
332 GLX_ACCUM_ALPHA_SIZE, format.accumulationBufferAlphaBits
333 };
334
335 allAttribs.insert (allAttribs.end(), optionalAttribs.begin(), optionalAttribs.end());
336
337 allAttribs.push_back (None);
338
339 int nElements = 0;
340 bestConfig = makeXFreePtr (glXChooseFBConfig (display, X11Symbols::getInstance()->xDefaultScreen (display), allAttribs.data(), &nElements));
341
342 return nElements != 0 && bestConfig != nullptr;
343 }
