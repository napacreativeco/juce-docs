94 template <typename ProcessContext>
95 void process (const ProcessContext& context) noexcept
96 {
97 const auto& inputBlock = context.getInputBlock();
98 auto& outputBlock = context.getOutputBlock();
99 const auto numChannels = outputBlock.getNumChannels();
100 const auto numSamples = outputBlock.getNumSamples();
101
102 jassert (inputBlock.getNumChannels() <= s1.size());
103 jassert (inputBlock.getNumChannels() == numChannels);
104 jassert (inputBlock.getNumSamples() == numSamples);
105
106 if (context.isBypassed)
107 {
108 outputBlock.copyFrom (inputBlock);
109 return;
110 }
111
112 for (size_t channel = 0; channel < numChannels; ++channel)
113 {
114 auto* inputSamples = inputBlock .getChannelPointer (channel);
115 auto* outputSamples = outputBlock.getChannelPointer (channel);
116
117 for (size_t i = 0; i < numSamples; ++i)
118 outputSamples[i] = processSample ((int) channel, inputSamples[i]);
119 }
120
121 #if JUCE_DSP_ENABLE_SNAP_TO_ZERO
122 snapToZero();
123 #endif
124 }
