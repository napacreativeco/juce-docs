36class Quaternion

38public:
39 Quaternion() noexcept : scalar() {}
40 Quaternion (const Quaternion& other) noexcept : vector (other.vector), scalar (other.scalar) {}
41 Quaternion (Vector3D<Type> vectorPart, Type scalarPart) noexcept : vector (vectorPart), scalar (scalarPart) {}
42 Quaternion (Type x, Type y, Type z, Type w) noexcept : vector (x, y, z), scalar (w) {}

45 static Quaternion fromAngle (Type angle, Vector3D<Type> axis) noexcept
46 {
47 return Quaternion (axis.normalised() * std::sin (angle / (Type) 2), std::cos (angle / (Type) 2));
48 }
49
50 Quaternion& operator= (Quaternion other) noexcept
51 {
52 vector = other.vector;
53 scalar = other.scalar;
54 return *this;
55 }
56
57 Quaternion& operator*= (Quaternion other) noexcept
58 {
59 const Type oldScalar (scalar);
60 scalar = (scalar * other.scalar) - (vector * other.vector);
61 vector = (other.vector * oldScalar) + (vector * other.scalar) + (vector ^ other.vector);
62 return *this;
63 }
64
65 Type length() const noexcept { return std::sqrt (normal()); }
66 Type normal() const noexcept { return scalar * scalar + vector.lengthSquared(); }
67
68 Quaternion normalised() const noexcept
69 {
70 const Type len (length());
71 jassert (len > 0);
72 return Quaternion (vector / len, scalar / len);
73 }

76 Matrix3D<Type> getRotationMatrix() const noexcept
77 {
78 const Type norm (normal());
79 const Type s (norm > 0 ? ((Type) 2) / norm : 0);
80 const Type xs (s * vector.x), ys (s * vector.y), zs (s * vector.z);
81 const Type wx (xs * scalar), wy (ys * scalar), wz (zs * scalar);
82 const Type xx (xs * vector.x), xy (ys * vector.x), xz (zs * vector.x);
83 const Type yy (ys * vector.y), yz (zs * vector.y), zz (zs * vector.z);
84
85 return Matrix3D<Type> (((Type) 1) - (yy + zz), xy - wz, xz + wy, 0,
86 xy + wz, ((Type) 1) - (xx+ zz), yz - wx, 0,
87 xz - wy, yz + wx, ((Type) 1) - (xx + yy), 0,
88 0, 0, 0, (Type) 1);
89 }
