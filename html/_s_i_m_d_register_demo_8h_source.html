56template <typename T>
57static T* toBasePointer (SIMDRegister<T>* r) noexcept
58{
59 return reinterpret_cast<T*> (r);
60}

62constexpr auto registerSize = dsp::SIMDRegister<float>::size();
63
64//==============================================================================
65struct SIMDRegisterDemoDSP
66{
67 void prepare (const ProcessSpec& spec)
68 {
69 sampleRate = spec.sampleRate;
70
71 iirCoefficients = IIR::Coefficients<float>::makeLowPass (sampleRate, 440.0f);
72 iir.reset (new IIR::Filter<SIMDRegister<float>> (iirCoefficients));
73
74 interleaved = AudioBlock<SIMDRegister<float>> (interleavedBlockData, 1, spec.maximumBlockSize);
75 zero = AudioBlock<float> (zeroData, SIMDRegister<float>::size(), spec.maximumBlockSize);
76
77 zero.clear();
78
79 auto monoSpec = spec;
80 monoSpec.numChannels = 1;
81 iir->prepare (monoSpec);
82 }
83
84 template <typename SampleType>
85 auto prepareChannelPointers (const AudioBlock<SampleType>& block)
86 {
87 std::array<SampleType*, registerSize> result {};
88
89 for (size_t ch = 0; ch < result.size(); ++ch)
90 result[ch] = (ch < block.getNumChannels() ? block.getChannelPointer (ch) : zero.getChannelPointer (ch));
91
92 return result;
93 }
94
95 void process (const ProcessContextReplacing<float>& context)
96 {
97 jassert (context.getInputBlock().getNumSamples() == context.getOutputBlock().getNumSamples());
98 jassert (context.getInputBlock().getNumChannels() == context.getOutputBlock().getNumChannels());
99
100 const auto& input = context.getInputBlock();
101 const auto numSamples = (int) input.getNumSamples();
102
103 auto inChannels = prepareChannelPointers (input);
104
105 using Format = AudioData::Format<AudioData::Float32, AudioData::NativeEndian>;
106
107 AudioData::interleaveSamples (AudioData::NonInterleavedSource<Format> { inChannels.data(), registerSize, },
108 AudioData::InterleavedDest<Format> { toBasePointer (interleaved.getChannelPointer (0)), registerSize },
109 numSamples);
110
111 iir->process (ProcessContextReplacing<SIMDRegister<float>> (interleaved));
112
113 auto outChannels = prepareChannelPointers (context.getOutputBlock());
114
115 AudioData::deinterleaveSamples (AudioData::InterleavedSource<Format> { toBasePointer (interleaved.getChannelPointer (0)), registerSize },
116 AudioData::NonInterleavedDest<Format> { outChannels.data(), registerSize },
117 numSamples);
118 }
119
120 void reset()
121 {
122 iir.reset();
123 }
124
125 void updateParameters()
126 {
127 if (sampleRate != 0.0)
128 {
129 auto cutoff = static_cast<float> (cutoffParam.getCurrentValue());
130 auto qVal = static_cast<float> (qParam.getCurrentValue());
131
132 switch (typeParam.getCurrentSelectedID())
133 {
134 case 1: *iirCoefficients = IIR::ArrayCoefficients<float>::makeLowPass (sampleRate, cutoff, qVal); break;
135 case 2: *iirCoefficients = IIR::ArrayCoefficients<float>::makeHighPass (sampleRate, cutoff, qVal); break;
136 case 3: *iirCoefficients = IIR::ArrayCoefficients<float>::makeBandPass (sampleRate, cutoff, qVal); break;
137 default: break;
138 }
139 }
140 }
141
142 //==============================================================================
143 IIR::Coefficients<float>::Ptr iirCoefficients;
144 std::unique_ptr<IIR::Filter<SIMDRegister<float>>> iir;
145
146 AudioBlock<SIMDRegister<float>> interleaved;
147 AudioBlock<float> zero;
148
149 HeapBlock<char> interleavedBlockData, zeroData;
150
151 ChoiceParameter typeParam { { "Low-pass", "High-pass", "Band-pass" }, 1, "Type" };
152 SliderParameter cutoffParam { { 20.0, 20000.0 }, 0.5, 440.0f, "Cutoff", "Hz" };
153 SliderParameter qParam { { 0.3, 20.0 }, 0.5, 0.7, "Q" };
154
155 std::vector<DSPDemoParameterBase*> parameters { &typeParam, &cutoffParam, &qParam };
156 double sampleRate = 0.0;
157};
